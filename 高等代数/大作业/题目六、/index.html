    题目六、

    ![[b8eb137017dab314763a782f556ad4b1.jpeg]]好，这个作业要求你做这几件事：

    1. 数据获取

    找或采集一个评分矩阵，缺失率>80%

    • 公开数据集：MovieLens、Book-Crossing、Amazon评论
    • 或者自己设计问卷采集校园设施/食堂评价

    2. 矩阵补全

    用低秩分解方法填补缺失值

    • 常用方法：SVD截断、矩阵分解（MF）、交替最小二乘（ALS）
    • 核心思想：假设矩阵是低秩的，用已观测值推断缺失值

    3. 解释潜在特征

    分解后得到 U（用户特征矩阵）和 V（物品特征矩阵）

    • 每一列是一个"潜在特征"
    • 你需要解释：这些特征代表什么？（比如"动作片偏好"、"价格敏感度"等）

    4. 数值验证

    证明你的补全是有意义的

    • 训练集/测试集划分
    • 预测误差（RMSE/MAE）

    1、数据获取¶

    找到了一个已经压缩过了的数据量适中（10000×4499）的Netflix Prize精简版，每行每列分别是一个用户和一个电影![[Pasted image 20260415193223.png]]，这是一个截面

    📊 列名含义¶

    列名 含义
    X1 第1部电影
    X2 第2部电影
    X3 第3部电影
    ... ...
    X4499 第4499部电影

    📈 数据值含义¶

    值 含义
    0 用户未评分（缺失值）
    1 1星
    2 2星
    3 3星
    4 4星
    5 5星

    💡 关键点¶

    • 这是Netflix原始评分精简版，评分范围是1-5星
    • 0不代表评分为0，而是"缺失"——用户没看过或没评分
    • 数据已经整理成矩阵形式：10000行（用户）× 4499列（电影）
    • 稀疏度极高，大部分是0

    处理时记得把0当成缺失值，不要当评分用。

    MF补全的核心思路¶

    训练阶段：只用已观测值优化 P 和 Q

    补全阶段：用训练好的 P 和 Q 预测所有位置

    $\(R_{补全} = P \cdot Q^T\)$ python实现

      import numpy as np
import pandas as pd             

def LinearRegression(data, hide = 10, l2_lambda = 0.02, learning_rate = 0.02, epochs = 1000):
        m, n = data.shape
        P = np.random.randn(m, hide) * 0.01
        Q = np.random.randn(n, hide) * 0.01
        for num in range(epochs):
            for i in range(m):
                for j in range(n):
                    if data[i, j]:
                        error = data[i, j] - P[i, :] @ Q.T[:, j]
                        P[i, :] = P[i, :] + learning_rate * (error*Q.T[:, j] - l2_lambda*P[i, :])
                        Q[j, :] = Q[j, :] + learning_rate * (error*P[i, :] - l2_lambda*Q[j, :])

        return P, Q, P@Q.T

# 读取数据
data = pd.read_csv("database.csv").values
result_P, result_Q, result = LinearRegression(data)

result_P.to_csv("补全矩阵P.csv", index=False)
result_Q.to_csv("补全矩阵Q.csv", index=False)
result.to_csv("补全矩阵.csv", index=False)

      import numpy as np
import pandas as pd

def ALS(data, k=10, l2_lambda=0.02, epochs=100):
    """
    交替最小二乘法 - 每步有闭式解，收敛快
    """
    m, n = data.shape
    P = np.random.randn(m, k) * 0.01
    Q = np.random.randn(n, k) * 0.01

    mask = ~np.isnan(data)

    for epoch in range(epochs):
        # 固定 Q，更新 P
        for i in range(m):
            mask_i = mask[i, :]  # 用户i评分过的物品
            if np.sum(mask_i) > 0:
                Q_i = Q[mask_i, :]
                r_i = data[i, mask_i]
                P[i, :] = np.linalg.solve(
                    Q_i.T @ Q_i + l2_lambda * np.eye(k),
                    Q_i.T @ r_i
                )

        # 固定 P，更新 Q
        for j in range(n):
            mask_j = mask[:, j]  # 评分过物品j的用户
            if np.sum(mask_j) > 0:
                P_j = P[mask_j, :]
                r_j = data[mask_j, j]
                Q[j, :] = np.linalg.solve(
                    P_j.T @ P_j + l2_lambda * np.eye(k),
                    P_j.T @ r_j
                )

        # 计算RMSE
        if (epoch + 1) % 10 == 0:
            pred = P @ Q.T
            rmse = np.sqrt(np.mean((data[mask] - pred[mask])**2))
            print(f"Epoch {epoch+1}, RMSE: {rmse:.4f}")

    return P, Q, P @ Q.T

data = pd.read_csv("database.csv").values
result_P, result_Q, result = ALS(data, k=10, epochs=100)

np.savetxt("补全矩阵P.csv", result_P, delimiter=",")
np.savetxt("补全矩阵Q.csv", result_Q, delimiter=",")
np.savetxt("补全矩阵.csv", result, delimiter=",")
