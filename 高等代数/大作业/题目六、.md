## MF补全的核心思路

训练阶段：只用**已观测值**优化 P 和 Q

补全阶段：用训练好的 P 和 Q **预测所有位置**

$$R_{补全} = P \cdot Q^T$$
python实现

1. 直接计算
```python
import numpy as np
import pandas as pd             

def LinearRegression(data, hide = 10, l2_lambda = 0.02, learning_rate = 0.02, epochs = 1000):
        m, n = data.shape
        P = np.random.randn(m, hide) * 0.01
        Q = np.random.randn(n, hide) * 0.01
        for num in range(epochs):
            for i in range(m):
                for j in range(n):
                    if data[i, j]:
                        error = data[i, j] - P[i, :] @ Q.T[:, j]
                        P[i, :] = P[i, :] + learning_rate * (error*Q.T[:, j] - l2_lambda*P[i, :])
                        Q[j, :] = Q[j, :] + learning_rate * (error*P[i, :] - l2_lambda*Q[j, :])

        return P, Q, P@Q.T

# 读取数据
data = pd.read_csv("database.csv").values
result_P, result_Q, result = LinearRegression(data)

result_P.to_csv("补全矩阵P.csv", index=False)
result_Q.to_csv("补全矩阵Q.csv", index=False)
result.to_csv("补全矩阵.csv", index=False)
```

2. ALS算法优化
```python
import numpy as np
import pandas as pd

def ALS(data, k=10, l2_lambda=0.02, epochs=100):
    """
    交替最小二乘法 - 每步有闭式解，收敛快
    """
    m, n = data.shape
    P = np.random.randn(m, k) * 0.01
    Q = np.random.randn(n, k) * 0.01
    
    mask = ~np.isnan(data)
    
    for epoch in range(epochs):
        # 固定 Q，更新 P
        for i in range(m):
            mask_i = mask[i, :]  # 用户i评分过的物品
            if np.sum(mask_i) > 0:
                Q_i = Q[mask_i, :]
                r_i = data[i, mask_i]
                P[i, :] = np.linalg.solve(
                    Q_i.T @ Q_i + l2_lambda * np.eye(k),
                    Q_i.T @ r_i
                )
        
        # 固定 P，更新 Q
        for j in range(n):
            mask_j = mask[:, j]  # 评分过物品j的用户
            if np.sum(mask_j) > 0:
                P_j = P[mask_j, :]
                r_j = data[mask_j, j]
                Q[j, :] = np.linalg.solve(
                    P_j.T @ P_j + l2_lambda * np.eye(k),
                    P_j.T @ r_j
                )
        
        # 计算RMSE
        if (epoch + 1) % 10 == 0:
            pred = P @ Q.T
            rmse = np.sqrt(np.mean((data[mask] - pred[mask])**2))
            print(f"Epoch {epoch+1}, RMSE: {rmse:.4f}")
    
    return P, Q, P @ Q.T

data = pd.read_csv("database.csv").values
result_P, result_Q, result = ALS(data, k=10, epochs=100)

np.savetxt("补全矩阵P.csv", result_P, delimiter=",")
np.savetxt("补全矩阵Q.csv", result_Q, delimiter=",")
np.savetxt("补全矩阵.csv", result, delimiter=",")
```
