# Function to match cluster labels to true labels for evaluation
def match_labels(true_labels, cluster_labels):
    cost_matrix = confusion_matrix(true_labels, cluster_labels)
    row_ind, col_ind = linear_sum_assignment(cost_matrix, maximize=True)
    label_mapping = dict(zip(col_ind, row_ind))
    mapped_labels = [label_mapping.get(label, -1) for label in cluster_labels]
    return mapped_labels, cost_matrix
