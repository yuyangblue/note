## 一、生成拟合数据

首先，我们来造一批数据——用 `sin(2πx)` 加点噪声，这样才有"真实感"。

```python
import numpy as np
import matplotlib.pyplot as plt

# 设置随机种子，保证结果可复现
np.random.seed(42)

# 生成训练数据
N = 20  # 样本数量
x_train = np.random.uniform(0, 1, N)  # 在[0,1]区间均匀采样
t_train = np.sin(2 * np.pi * x_train) + np.random.normal(0, 0.2, N)  # sin曲线 + 高斯噪声

# 生成测试数据（更密，用于评估）
x_test = np.linspace(0, 1, 100)
t_test = np.sin(2 * np.pi * x_test)

# 画个图看看
plt.figure(figsize=(10, 4))
plt.scatter(x_train, t_train, color='blue', alpha=0.6, label='训练数据')
plt.plot(x_test, t_test, 'r--', label='真实函数')
plt.legend()
plt.title('线性回归示例数据')
plt.xlabel('x')
plt.ylabel('t')
plt.show()
```

**为什么这样写？**
- `np.random.uniform(0, 1, N)` 生成均匀分布的 x，比随机采样更适合展示拟合效果
- 加噪声模拟真实场景中的测量误差

## 二、最小二乘法：正规方程求解

笔记里讲的核心公式：`w* = (Z^T Z)^(-1) Z^T t`

这里 Z 是设计矩阵，存放基函数转换后的特征。

```python
def design_matrix(x, M, basis_type='polynomial'):
    """
    构建设计矩阵 Z
    
    参数:
        x: 输入数据，形状 (N,)
        M: 基函数个数（多项式最高次+1）
        basis_type: 基函数类型
    
    返回:
        Z: 设计矩阵，形状 (N, M)
    """
    N = len(x)
    Z = np.zeros((N, M))
    
    if basis_type == 'polynomial':
        # 多项式基函数: [1, x, x^2, ..., x^(M-1)]
        for j in range(M):
            Z[:, j] = x ** j
    
    elif basis_type == 'gaussian_rbf':
        # 高斯径向基函数: exp(-(x - mu_j)^2 / (2*sigma^2))
        # mu_j 在 [0, 1] 区间均匀分布
        mu = np.linspace(0, 1, M)
        sigma = 0.1
        for j in range(M):
            Z[:, j] = np.exp(-(x - mu[j])**2 / (2 * sigma**2))
    
    return Z

# 用正规方程求解最优参数
def least_squares_fit(x, t, M, basis_type='polynomial', lambda_reg=0):
    """
    最小二乘法求解线性回归
    
    参数:
        x, t: 训练数据
        M: 基函数个数
        lambda_reg: L2正则化系数（可选，防止矩阵奇异）
    
    返回:
        w: 最优参数
    """
    Z = design_matrix(x, M, basis_type)
    
    # 正规方程: w* = (Z^T Z + lambda*I)^(-1) Z^T t
    # 加上小的正则项 lambda_reg 确保矩阵可逆
    if lambda_reg > 0:
        ZTZ = Z.T @ Z + lambda_reg * np.eye(M)
    else:
        ZTZ = Z.T @ Z
    
    Zt = Z.T @ t
    w = np.linalg.solve(ZTZ, Zt)  # 比求逆更数值稳定
    
    return w

# 预测函数
def predict(x, w, basis_type='polynomial'):
    """用学到的参数做预测"""
    M = len(w)
    Z = design_matrix(x, M, basis_type)
    return Z @ w
```

**为什么这样写？**
- `np.linalg.solve(A, b)` 比 `np.linalg.inv(A) @ b` 更稳定，避免了求逆的数值问题
- 正则化项 `lambda_reg * np.eye(M)` 保证矩阵正定可逆

## 三、不同多项式次数的拟合效果

看看 M=1, 3, 10 时模型的表现——直观理解过拟合！

```python
# 测试不同多项式次数
degrees = [1, 3, 10]
plt.figure(figsize=(12, 4))

for i, M in enumerate(degrees):
    # 训练
    w = least_squares_fit(x_train, t_train, M, basis_type='polynomial')
    
    # 预测
    x_plot = np.linspace(0, 1, 200)
    y_plot = predict(x_plot, w, basis_type='polynomial')
    
    # 计算训练误差
    y_train_pred = predict(x_train, w, basis_type='polynomial')
    train_error = np.mean((y_train_pred - t_train)**2)
    
    # 画图
    plt.subplot(1, 3, i+1)
    plt.scatter(x_train, t_train, color='blue', alpha=0.5, label='训练数据')
    plt.plot(x_plot, y_plot, 'r-', label=f'M={M}')
    plt.plot(x_test, t_test, 'g--', alpha=0.5, label='真实函数')
    plt.title(f'多项式 M={M}, 训练误差={train_error:.4f}')
    plt.legend()
    plt.ylim(-1.5, 1.5)

plt.tight_layout()
plt.show()

# 打印参数，观察过拟合时参数的 magnitude
print("参数幅度对比:")
for M in degrees:
    w = least_squares_fit(x_train, t_train, M, basis_type='polynomial')
    print(f"  M={M}: max|w| = {np.max(np.abs(w)):.2f}")
```

**输出示例：**
```
参数幅度对比:
  M=1: max|w| = 0.76
  M=3: max|w| = 3.41
  M=10: max|w| = 156.23  # 参数爆炸！
```

## 四、高斯 RBF 基函数扩展

笔记里提到的"非线性基函数"，高斯 RBF 是很常用的一种。

```python
# 用高斯RBF基函数，M=9
M_rbf = 9
w_rbf = least_squares_fit(x_train, t_train, M_rbf, basis_type='gaussian_rbf')

# 预测
x_plot = np.linspace(0, 1, 200)
y_rbf = predict(x_plot, w_rbf, basis_type='gaussian_rbf')

plt.figure(figsize=(8, 5))
plt.scatter(x_train, t_train, color='blue', alpha=0.6, label='训练数据')
plt.plot(x_test, t_test, 'g--', label='真实函数')
plt.plot(x_plot, y_rbf, 'r-', linewidth=2, label='RBF拟合')
plt.legend()
plt.title(f'高斯RBF基函数 (M={M_rbf})')
plt.show()
```

**为什么这样写？**
- 每个高斯基函数的中心 `mu` 均匀分布在输入区间
- σ 控制基函数的宽度，太大则所有基函数几乎相同，太小则稀疏

## 五、带正则化的岭回归

防止过拟合！正则化项让参数不会爆炸。

```python
# 测试不同正则化强度
lambda_values = [0, 0.001, 0.01, 0.1, 1.0]
M = 10  # 高次多项式，容易过拟合

plt.figure(figsize=(12, 4))

for i, lam in enumerate(lambda_values):
    # 带正则化的拟合
    w_reg = least_squares_fit(x_train, t_train, M, 
                              basis_type='polynomial', 
                              lambda_reg=lam)
    
    # 预测
    y_reg = predict(x_plot, w_reg, basis_type='polynomial')
    
    # 计算误差
    y_train_reg = predict(x_train, w_reg, basis_type='polynomial')
    train_err = np.mean((y_train_reg - t_train)**2)
    
    plt.subplot(1, 5, i+1)
    plt.scatter(x_train, t_train, color='blue', alpha=0.5)
    plt.plot(x_plot, y_reg, 'r-')
    plt.plot(x_test, t_test, 'g--', alpha=0.5)
    plt.title(f'λ={lam}\nerr={train_err:.3f}')
    plt.ylim(-1.5, 1.5)

plt.tight_layout()
plt.show()

# 对比参数幅度
print("\n正则化对参数幅度的影响 (M=10):")
for lam in lambda_values:
    w = least_squares_fit(x_train, t_train, 10, lambda_reg=lam)
    print(f"  λ={lam:5.3f}: max|w| = {np.max(np.abs(w)):.2f}")
```

## 六、完整示例：sklearn 风格封装

把上面的代码封装成 sklearn 风格的类，方便调用。

```python
class LinearRegression:
    """线性回归模型，支持多种基函数和正则化"""
    
    def __init__(self, M=3, basis='polynomial', lambda_reg=0):
        self.M = M
        self.basis = basis
        self.lambda_reg = lambda_reg
        self.w = None
    
    def fit(self, X, y):
        """训练模型"""
        # X 可能是 (N,) 或 (N,1)，统一处理
        if X.ndim == 1:
            X = X.reshape(-1, 1)
        x = X.flatten()
        
        self.w = least_squares_fit(x, y, self.M, self.basis, self.lambda_reg)
        return self
    
    def predict(self, X):
        """预测"""
        if X.ndim == 1:
            X = X.reshape(-1, 1)
        x = X.flatten()
        return predict(x, self.w, self.basis)

# 使用示例
model = LinearRegression(M=5, basis='polynomial', lambda_reg=0.01)
model.fit(x_train, t_train)
y_pred = model.predict(x_test)

plt.figure()
plt.scatter(x_train, t_train, label='训练数据')
plt.plot(x_test, y_pred, 'r-', label='模型预测')
plt.plot(x_test, t_test, 'g--', alpha=0.5, label='真实函数')
plt.legend()
plt.show()
```

## 七、关键知识点回顾

| 概念 | 代码实现 | 要点 |
|------|---------|------|
| 正规方程 | `w = solve(Z.T @ Z, Z.T @ t)` | 解析解，一步到位 |
| 多项式基 | `Z[:, j] = x ** j` | M=1是直线，M越大越灵活 |
| RBF基函数 | `exp(-(x-mu)²/2σ²)` | 局部作用，天然平滑 |
| 正则化 | `Z.T @ Z + λI` | 防止参数爆炸，改善病态 |
| 预测 | `y = Z @ w` | 矩阵乘法，并行高效 |
