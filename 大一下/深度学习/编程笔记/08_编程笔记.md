## 一、分类问题评价指标

准确率只是冰山一角，让我们全面认识评价指标。

```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix, roc_curve, auc, precision_recall_curve, average_precision_score
from sklearn.model_selection import cross_val_score, StratifiedKFold
from sklearn.datasets import make_classification, make_moons
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.neural_network import MLPClassifier
from sklearn.preprocessing import StandardScaler
import warnings
warnings.filterwarnings('ignore')

# 生成二分类数据（有一定重叠，更真实）
np.random.seed(42)
X, y = make_classification(
    n_samples=500, n_features=20, n_informative=15,
    n_redundant=5, n_classes=2, flip_y=0.1,
    class_sep=0.8, random_state=42
)

# 数据标准化
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

print("数据集信息:")
print(f"  样本数: {len(y)}")
print(f"  特征数: {X.shape[1]}")
print(f"  正类比例: {np.mean(y):.2%}")

# ============================================
# 1. 混淆矩阵
# ============================================
def plot_confusion_matrix(y_true, y_pred, labels=['负类', '正类']):
    """绘制混淆矩阵"""
    cm = confusion_matrix(y_true, y_pred)
    
    fig, ax = plt.subplots(figsize=(6, 5))
    im = ax.imshow(cm, cmap='Blues')
    
    # 添加数值标签
    for i in range(2):
        for j in range(2):
            text = ax.text(j, i, cm[i, j], ha='center', va='center',
                         fontsize=16, color='white' if cm[i, j] > cm.max()/2 else 'black')
    
    ax.set_xticks([0, 1])
    ax.set_yticks([0, 1])
    ax.set_xticklabels(labels)
    ax.set_yticklabels(labels)
    ax.set_xlabel('预测标签')
    ax.set_ylabel('真实标签')
    ax.set_title('混淆矩阵')
    
    # 添加颜色条
    plt.colorbar(im, ax=ax)
    plt.tight_layout()
    plt.show()
    
    # 计算指标
    TN, FP, FN, TP = cm.ravel()
    print(f"\n混淆矩阵解读:")
    print(f"  TN (真负): {TN}, FP (假正): {FP}")
    print(f"  FN (假负): {FN}, TP (真正): {TP}")
    
    accuracy = (TP + TN) / (TP + TN + FP + FN)
    precision = TP / (TP + FP) if (TP + FP) > 0 else 0
    recall = TP / (TP + FN) if (TP + FN) > 0 else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0
    
    print(f"\n评价指标:")
    print(f"  准确率 (Accuracy):  {accuracy:.4f}")
    print(f"  精确率 (Precision): {precision:.4f}")
    print(f"  召回率 (Recall):    {recall:.4f}")
    print(f"  F1 分数:            {f1:.4f}")
    
    return accuracy, precision, recall, f1

# 训练模型
from sklearn.model_selection import train_test_split
X_train, X_test, y_train, y_test = train_test_split(
    X_scaled, y, test_size=0.3, random_state=42, stratify=y
)

model = LogisticRegression(max_iter=1000)
model.fit(X_train, y_train)
y_pred = model.predict(X_test)
y_pred_proba = model.predict_proba(X_test)[:, 1]

print("\n" + "="*50)
print("1. 混淆矩阵分析")
print("="*50)
accuracy, precision, recall, f1 = plot_confusion_matrix(y_test, y_pred)
```

## 二、ROC 曲线与 AUC

ROC 曲线展示模型在不同阈值下的表现，AUC 是 ROC 曲线下的面积。

```python
# ============================================
# 2. ROC 曲线与 AUC
# ============================================
print("\n" + "="*50)
print("2. ROC 曲线与 AUC")
print("="*50)

def plot_roc_curves(models_predictions, model_names):
    """绘制多条 ROC 曲线"""
    plt.figure(figsize=(8, 6))
    
    for (y_true, y_score), name in zip(models_predictions, model_names):
        fpr, tpr, thresholds = roc_curve(y_true, y_score)
        roc_auc = auc(fpr, tpr)
        
        plt.plot(fpr, tpr, linewidth=2, label=f'{name} (AUC = {roc_auc:.3f})')
    
    plt.plot([0, 1], [0, 1], 'k--', alpha=0.5, label='随机猜测')
    plt.xlabel('假正率 (FPR)')
    plt.ylabel('真正率 (TPR)')
    plt.title('ROC 曲线对比')
    plt.legend(loc='lower right')
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.show()

# 对比多个模型
models = {
    'Logistic Regression': LogisticRegression(max_iter=1000),
    'Random Forest': RandomForestClassifier(n_estimators=100, random_state=42),
    'Neural Network': MLPClassifier(hidden_layer_sizes=(50, 25), max_iter=500, random_state=42)
}

predictions = []
for name, clf in models.items():
    clf.fit(X_train, y_train)
    y_pred = clf.predict(X_test)
    y_pred_proba = clf.predict_proba(X_test)[:, 1]
    predictions.append((y_test, y_pred_proba))
    
    # 计算 AUC
    fpr, tpr, _ = roc_curve(y_test, y_pred_proba)
    model_auc = auc(fpr, tpr)
    print(f"  {name}: AUC = {model_auc:.4f}")

plot_roc_curves(predictions, list(models.keys()))

# ============================================
# AUC 的直观理解
# ============================================
print("\nAUC 的直观理解:")
print("  - AUC = 1.0: 完美分类器")
print("  - AUC = 0.5: 等价于随机猜测")
print("  - AUC < 0.5: 比随机还差（预测反转）")
```

## 三、Precision-Recall 曲线

当类别不平衡时，PR 曲线比 ROC 更有意义。

```python
# ============================================
# 3. Precision-Recall 曲线
# ============================================
print("\n" + "="*50)
print("3. Precision-Recall 曲线")
print("="*50)

def plot_pr_curve(y_true, y_score, model_name):
    """绘制 PR 曲线"""
    precision, recall, thresholds = precision_recall_curve(y_true, y_score)
    ap = average_precision_score(y_true, y_score)
    
    plt.figure(figsize=(8, 6))
    plt.plot(recall, precision, linewidth=2, label=f'{model_name} (AP = {ap:.3f})')
    plt.xlabel('召回率 (Recall)')
    plt.ylabel('精确率 (Precision)')
    plt.title(f'Precision-Recall 曲线: {model_name}')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.show()
    
    print(f"  Average Precision (AP) = {ap:.4f}")

# Logistic Regression 的 PR 曲线
plot_pr_curve(y_test, y_pred_proba, 'Logistic Regression')

# ============================================
# 极端不平衡情况演示
# ============================================
print("\n极端不平衡情况:")
# 创建高度不平衡的数据
X_im, y_im = make_classification(
    n_samples=1000, n_features=20, n_informative=10,
    n_classes=2, weights=[0.95, 0.05],  # 95%负类，5%正类
    random_state=42
)

X_train_im, X_test_im, y_train_im, y_test_im = train_test_split(
    X_im, y_im, test_size=0.3, random_state=42, stratify=y_im
)

model_im = LogisticRegression(max_iter=1000)
model_im.fit(X_train_im, y_train_im)
y_pred_im = model_im.predict(X_test_im)
y_pred_proba_im = model_im.predict_proba(X_test_im)[:, 1]

# ROC vs PR
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

# ROC
fpr, tpr, _ = roc_curve(y_test_im, y_pred_proba_im)
axes[0].plot(fpr, tpr, linewidth=2)
axes[0].plot([0, 1], [0, 1], 'k--')
axes[0].set_xlabel('FPR')
axes[0].set_ylabel('TPR')
axes[0].set_title(f'ROC (AUC={auc(fpr, tpr):.3f})')
axes[0].grid(True, alpha=0.3)

# PR
precision, recall, _ = precision_recall_curve(y_test_im, y_pred_proba_im)
axes[1].plot(recall, precision, linewidth=2)
axes[1].set_xlabel('Recall')
axes[1].set_ylabel('Precision')
axes[1].set_title(f'PR (AP={average_precision_score(y_test_im, y_pred_proba_im):.3f})')
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

print(f"  正类比例: {np.mean(y_test_im):.2%}")
print("  → PR 曲线更能反映不平衡数据的表现！")
```

## 四、交叉验证

别让数据集划分毁了你的结论！

```python
# ============================================
# 4. 交叉验证
# ============================================
print("\n" + "="*50)
print("4. 交叉验证")
print("="*50)

def manual_cross_val(model, X, y, k=5, random_state=42):
    """手动实现 K 折交叉验证"""
    np.random.seed(random_state)
    n = len(y)
    indices = np.arange(n)
    np.random.shuffle(indices)
    
    # 分成 k 份
    fold_size = n // k
    scores = []
    
    for i in range(k):
        # 第 i 折作为测试集
        start = i * fold_size
        end = start + fold_size if i < k - 1 else n
        
        test_idx = indices[start:end]
        train_idx = np.concatenate([indices[:start], indices[end:]])
        
        X_train_fold = X[train_idx]
        y_train_fold = y[train_idx]
        X_test_fold = X[test_idx]
        y_test_fold = y[test_idx]
        
        # 训练和评估
        model_clone = LogisticRegression(max_iter=1000)  # 重新实例化
        model_clone.fit(X_train_fold, y_train_fold)
        score = model_clone.score(X_test_fold, y_test_fold)
        scores.append(score)
    
    return np.array(scores)

# 使用 sklearn 的交叉验证
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

print("\n使用 sklearn 的交叉验证:")
scores_sklearn = cross_val_score(

print("\n手动实现交叉验证:")

# 左图：各折得分
```

## 五、过拟合与欠拟合诊断

学习曲线告诉你模型到底怎么了。

```python
# ============================================
# 5. 学习曲线：诊断过拟合/欠拟合
# ============================================
print("\n" + "="*50)
print("5. 学习曲线诊断")
print("="*50)

from sklearn.model_selection import learning_curve

def plot_learning_curve(estimator, X, y, title):
    """绘制学习曲线"""
    train_sizes, train_scores, val_scores = learning_curve(
        estimator, X, y, cv=5, 
        train_sizes=np.linspace(0.1, 1.0, 10),
        scoring='accuracy',
        n_jobs=-1
    )
    
    train_mean = train_scores.mean(axis=1)
    train_std = train_scores.std(axis=1)
    val_mean = val_scores.mean(axis=1)
    val_std = val_scores.std(axis=1)
    
    plt.figure(figsize=(8, 5))
    plt.plot(train_sizes, train_mean, 'b-', label='训练集', linewidth=2)
    plt.fill_between(train_sizes, train_mean - train_std, train_mean + train_std, 
                    alpha=0.2, color='blue')
    plt.plot(train_sizes, val_mean, 'r-', label='验证集', linewidth=2)
    plt.fill_between(train_sizes, val_mean - val_std, val_mean + val_std, 
                    alpha=0.2, color='red')
    
    plt.xlabel('训练样本数')
    plt.ylabel('准确率')
    plt.title(f'学习曲线: {title}')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.show()

# 欠拟合示例（太简单的模型）
print("\n诊断 1: 欠拟合（模型太简单）")
simple_model = LogisticRegression(max_iter=1000)
plot_learning_curve(simple_model, X_scaled, y, '欠拟合模型')

# 过拟合示例（太复杂的模型）
print("\n诊断 2: 过拟合（模型太复杂）")
complex_model = MLPClassifier(hidden_layer_sizes=(100, 100, 100), 
                             max_iter=500, random_state=42)
plot_learning_curve(complex_model, X_scaled, y, '过拟合模型')

# 诊断结论
print("\n诊断结论:")
print("  欠拟合: 训练集和验证集准确率都低，且差距小")
print("  过拟合: 训练集准确率高，验证集低，差距大")
print("  良好:   两者的差距小，且都收敛到较高值")
```

## 六、关键知识点回顾

| 指标 | 公式 | 适用场景 |
|------|------|---------|
| Accuracy | (TP+TN)/(TP+TN+FP+FN) | 平衡数据 |
| Precision | TP/(TP+FP) | 预测正类要准 |
| Recall | TP/(TP+FN) | 找出所有正类 |
| F1 Score | 2PR/(P+R) | 平衡 P 和 R |
| AUC-ROC | ROC 曲线下面积 | 排序能力 |
| AP (PR-AUC) | PR 曲线下面积 | 不平衡数据 |
