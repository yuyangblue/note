## 实验一：RISC-V 简易汇编器

### 一、题目描述

设计一个简单的汇编器，将 RISC-V 汇编语言翻译成 32 位二进制机器码。

**输入**：若干行汇编代码（每行一条指令）  
**输出**：对应的 32 位二进制字符串（每行一条）

**支持的指令**：
- `add rd, rs1, rs2`
- `sub rd, rs1, rs2`
- `addi rd, rs1, imm`
- `lw rd, imm(rs1)`
- `sw rs2, imm(rs1)`
- `beq rs1, rs2, LABEL`（标签跳转）

**要求**：
- 寄存器编号：x0 ~ x31，也支持别名（如 a0, t0 等）
- 立即数/偏移：十进制整数
- 输出 32 位二进制字符串，无空格，无 `0b` 前缀
- 标签行（如 `LABEL:`）不输出机器码

**样例**：
- 输入：`add x1, x2, x3`
- 输出：`00000000001100010000000010110011`

---

### 二、思路指导

#### 2.1 整体思路：两遍扫描法

汇编器不能看到一行就翻译一行，因为像 `beq x1, x2, LOOP` 这样的跳转指令，后面的 `LOOP` 标签可能还没定义。所以采用**两遍扫描**：

1. **第一遍**：从头到尾读一遍，记录所有 `标签:` 出现在第几行（第0行、第1行……）
2. **第二遍**：再读一遍，真正翻译指令。遇到标签就查表算偏移量

#### 2.2 指令格式：RISC-V 的几种类型

RISC-V 的指令按格式分为几种，我们要把不同的字段拼成 32 位：

| 类型 | 指令举例 | 格式（从高位到低位） |
|------|---------|-------------------|
| **R型** | add, sub | `funct7(7) + rs2(5) + rs1(5) + funct3(3) + rd(5) + opcode(7)` |
| **I型** | addi, lw | `imm(12) + rs1(5) + funct3(3) + rd(5) + opcode(7)` |
| **S型** | sw | `imm[11:5](7) + rs2(5) + rs1(5) + funct3(3) + imm[4:0](5) + opcode(7)` |
| **B型** | beq | `imm[12](1)+imm[10:5](6) + rs2(5) + rs1(5) + funct3(3) + imm[4:1](4)+imm[11](1) + opcode(7)` |

**记忆口诀**：RISC-V 的 opcode 尾数有规律——R型是 `0110011`，I型是 `0010011` 或 `0000011`，S型是 `0100011`，B型是 `1100011`。

#### 2.3 关键难点

1. **负数转二进制（补码）**：如果立即数是负数，不能直接转二进制。要加上 `2^位数` 后再转。例如 -1 用 12 位表示：`4096 + (-1) = 4095`，二进制是 `111111111111`。

2. **B型（beq）的立即数编码很怪**：13 位立即数，但最低位永远是 0（因为指令地址是 2 字节对齐的），所以实际存的是 `imm[12:1]`。而且位的顺序是打乱的，需要按手册重新排列。

3. **标签偏移量计算**：`beq x1, x2, LOOP` 中，如果当前是第 3 行，`LOOP` 在第 7 行，那么偏移 = `(7 - 3) × 4`（每行一条指令，占 4 字节）。

4. **预处理输入**：输入有逗号、括号、空格，要先统一换成空格，方便用 `stringstream` 分割。

### 三、完整代码

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <sstream>
using namespace std;

// 将整数转换为指定位数的二进制字符串（支持负数补码）
string intToBin(int n, int bits) {
    if (n < 0) {
        n = (1 << bits) + n;  // 补码：加上 2^bits
    }
    string result;
    for (int i = bits - 1; i >= 0; i--) {
        result += ((n >> i) & 1) ? '1' : '0';
    }
    return result;
}

// 解析寄存器编号，支持 x0-x31 或别名（如 a0, t0）
int parseReg(string s) {
    if (s[0] == 'x') {
        return stoi(s.substr(1));
    }
    // 别名映射表
    map<string, int> regAlias = {
        {"zero", 0}, {"ra", 1}, {"sp", 2}, {"gp", 3}, {"tp", 4},
        {"t0", 5}, {"t1", 6}, {"t2", 7}, {"s0", 8}, {"fp", 8},
        {"s1", 9}, {"a0", 10}, {"a1", 11}, {"a2", 12}, {"a3", 13},
        {"a4", 14}, {"a5", 15}, {"a6", 16}, {"a7", 17}, {"s2", 18},
        {"s3", 19}, {"s4", 20}, {"s5", 21}, {"s6", 22}, {"s7", 23},
        {"s8", 24}, {"s9", 25}, {"s10", 26}, {"s11", 27}, {"t3", 28},
        {"t4", 29}, {"t5", 30}, {"t6", 31}
    };
    if (regAlias.count(s)) return regAlias[s];
    return stoi(s);  // 兜底
}

// 编码单条指令，currentLine 是当前行号（从0开始），labels 是标签表
string encodeInstruction(vector<string>& tokens, int currentLine, map<string, int>& labels) {
    string op = tokens[0];

    // ==================== R型：add, sub ====================
    if (op == "add" || op == "sub") {
        string rd  = intToBin(parseReg(tokens[1]), 5);
        string rs1 = intToBin(parseReg(tokens[2]), 5);
        string rs2 = intToBin(parseReg(tokens[3]), 5);
        string funct7 = (op == "add") ? "0000000" : "0100000";
        return funct7 + rs2 + rs1 + "000" + rd + "0110011";
    }

    // ==================== I型：addi ====================
    else if (op == "addi") {
        string rd  = intToBin(parseReg(tokens[1]), 5);
        string rs1 = intToBin(parseReg(tokens[2]), 5);
        string imm = intToBin(stoi(tokens[3]), 12);
        return imm + rs1 + "000" + rd + "0010011";
    }

    // ==================== I型：lw ====================
    // 格式：lw rd, offset(rs1)  预处理后变成：lw rd offset rs1
    else if (op == "lw") {
        string rd  = intToBin(parseReg(tokens[1]), 5);
        int imm = stoi(tokens[2]);
        string rs1 = intToBin(parseReg(tokens[3]), 5);
        string immBin = intToBin(imm, 12);
        return immBin + rs1 + "010" + rd + "0000011";
    }

    // ==================== S型：sw ====================
    // 格式：sw rs2, offset(rs1)  预处理后变成：sw rs2 offset rs1
    else if (op == "sw") {
        string rs2 = intToBin(parseReg(tokens[1]), 5);
        int imm = stoi(tokens[2]);
        string rs1 = intToBin(parseReg(tokens[3]), 5);
        string immBin = intToBin(imm, 12);
        // S型把12位立即数拆成高位7位和低位5位
        string immHigh = immBin.substr(0, 7);   // imm[11:5]
        string immLow  = immBin.substr(7, 5);    // imm[4:0]
        return immHigh + rs2 + rs1 + "010" + immLow + "0100011";
    }

    // ==================== B型：beq ====================
    // 注意：B型的立即数编码顺序是乱的，需要重新拼接
    else if (op == "beq") {
        string rs1 = intToBin(parseReg(tokens[1]), 5);
        string rs2 = intToBin(parseReg(tokens[2]), 5);

        // 计算偏移量：标签行号差 × 4
        int offset;
        if (labels.count(tokens[3])) {
            offset = (labels[tokens[3]] - currentLine) * 4;
        } else {
            offset = stoi(tokens[3]);
        }

        string imm = intToBin(offset, 13);  // 13位，最低位是0

        // B型格式：imm[12] | imm[10:5] | rs2 | rs1 | funct3 | imm[4:1] | imm[11] | opcode
        // imm[0] 始终是0，不参与编码
        string imm12  = imm.substr(0, 1);      // imm[12]
        string imm105 = imm.substr(2, 6);      // imm[10:5]
        string imm41  = imm.substr(8, 4);       // imm[4:1]
        string imm11  = imm.substr(1, 1);       // imm[11]

        return imm12 + imm105 + rs2 + rs1 + "000" + imm41 + imm11 + "1100011";
    }

    return "UNKNOWN_INSTRUCTION";
}

// 预处理：把逗号、括号换成空格，去掉注释
string preprocess(string s) {
    string result;
    for (char c : s) {
        if (c == ',' || c == '(' || c == ')') {
            result += ' ';
        } else if (c == '#') {
            break;  // # 后面是注释，直接忽略
        } else {
            result += c;
        }
    }
    return result;
}

int main() {
    vector<string> codes;
    map<string, int> labels;  // 标签名 -> 行号
    string line;

    // 读取所有代码行
    while (getline(cin, line)) {
        if (line.empty()) break;
        codes.push_back(line);
    }

    // ====== 第一遍扫描：记录所有标签的位置 ======
    for (int i = 0; i < codes.size(); i++) {
        string processed = preprocess(codes[i]);
        size_t colonPos = processed.find(':');
        if (colonPos != string::npos) {
            string label = processed.substr(0, colonPos);
            // 去掉标签前后的空格
            label.erase(0, label.find_first_not_of(" \t"));
            label.erase(label.find_last_not_of(" \t") + 1);
            labels[label] = i;  // 记录标签在第 i 行
        }
    }

    // ====== 第二遍扫描：真正翻译指令 ======
    for (int i = 0; i < codes.size(); i++) {
        string processed = preprocess(codes[i]);

        // 去掉标签部分
        size_t colonPos = processed.find(':');
        if (colonPos != string::npos) {
            processed = processed.substr(colonPos + 1);
        }

        // 用 stringstream 分割成单词
        stringstream ss(processed);
        string token;
        vector<string> tokens;
        while (ss >> token) {
            tokens.push_back(token);
        }

        if (tokens.empty()) continue;  // 空行或纯标签行，跳过

        // 编码并输出
        string machineCode = encodeInstruction(tokens, i, labels);
        cout << machineCode << endl;
    }

    return 0;
}
```

---

### 四、代码解析

1. **`intToBin` 函数**：核心工具函数。正数直接转二进制；负数用补码，即 `n + 2^bits`。例如 -3 用 5 位表示：`32 + (-3) = 29`，二进制 `11101`。

2. **`preprocess` 函数**：把输入标准化。`lw x1, 0(x2)` 变成 `lw x1 0 x2`，这样后面用 `stringstream` 就能按空格分割。

3. **两遍扫描**：第一遍只找标签（带冒号的行），第二遍才真正翻译。这是处理"先使用后定义"的唯一办法。

4. **B型立即数重组**：RISC-V 的 B型指令为了和 S型共享部分硬件，把立即数的位顺序打乱了。`imm[12]` 放最高位，`imm[10:5]` 接在后面，`imm[4:1]` 和 `imm[11]` 放在低位。这是手册规定，死记硬背即可。

## 实验二：RISC-V 简易模拟器

### 一、题目描述

实现一个简易 RISC-V 模拟器，支持 `add`、`sub`、`addi`、`lw`、`sw`、`beq`、`bne` 指令。

**输入**：若干行 32 位二进制机器码（每行一条）  
**输出**：每行执行结果，格式为：
```
PC=0x0|add x2, x0, x0|x2=0x0|null
```
用 `|` 分隔四部分：PC值、汇编指令、寄存器写入、内存写入。

**样例**：
- 输入：`00000000000000000000000100110011`（add x2, x0, x0）
- 输出：`PC=0x0|add x2, x0, x0|x2=0x0|null`

---

### 二、思路指导

#### 2.1 整体思路：取指 → 译码 → 执行 → 输出

模拟器就是一个循环：
1. 读一条二进制指令
2. 根据 **opcode（指令的最后 7 位）** 判断是什么类型
3. 按类型解析出 rd、rs1、rs2、imm 等字段
4. 执行运算（加减、访存、跳转）
5. 按格式输出结果
6. PC 更新，取下一条

#### 2.2 如何解析二进制？

RISC-V 是定长指令（32 位），各字段位置固定。例如：
- `opcode`：第 25~31 位（最后 7 位）
- `rd`：第 20~24 位（5 位）
- `funct3`：第 17~19 位（3 位）
- `rs1`：第 12~16 位（5 位）
- `rs2`：第 7~11 位（5 位）
- `funct7`：第 0~6 位（7 位）

用 `substr` 截取字符串，再用 `stoi(..., nullptr, 2)` 转成整数。

#### 2.3 立即数处理（符号扩展）

I型指令的 12 位立即数是有符号数。如果最高位是 1，表示负数。例如二进制 `111111111100`（12位），最高位是 1，真实值 = 二进制转十进制 - 4096 = -4。

**公式**：`value = 无符号值 - (最高位 ? 2^位数 : 0)`

#### 2.4 跳转指令的处理

`beq` 和 `bne` 会改变 PC。如果条件满足，PC 跳到目标地址；否则 PC+4。

注意：这里的 `p` 是**行号索引**（0, 1, 2...），不是字节地址。偏移量除以 4 就是行号偏移。

### 三、完整代码

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <sstream>
using namespace std;

// 辅助函数：寄存器名格式化输出（如 x2）
string x(int in) {
    return "x" + to_string(in);
}

// 整数转十六进制字符串（如 0x64）
string to_hex(int n) {
    stringstream ss;
    ss << hex << n;
    return "0x" + ss.str();
}

// 二进制字符串转整数
int to_int(string str) {
    return stoi(str, nullptr, 2);
}

int main() {
    int reg[32];           // 32个寄存器
    reg[0] = 0;            // x0 永远是 0
    vector<string> pc;     // 存储所有指令
    int mem[1024] = {0};   // 4KB 内存，按字（4字节）访问

    // 读取所有二进制指令
    for (string in; getline(cin, in);) {
        pc.push_back(in);
    }

    int p = 0;  // 当前执行到第几条指令（行号）

    while (p < pc.size()) {
        string code;  // 当前指令的助记符

        // ==================== R型：add, sub ====================
        // opcode = 0110011
        if (pc[p].substr(25, 7) == "0110011") {
            // 通过 funct7 区分 add 和 sub
            code = (pc[p].substr(0, 7) == "0000000") ? "add" : "sub";

            int rs2 = to_int(pc[p].substr(7, 5));   // 第2个源寄存器
            int rs1 = to_int(pc[p].substr(12, 5));  // 第1个源寄存器
            int rd  = to_int(pc[p].substr(20, 5));  // 目标寄存器

            if (code == "add") {
                reg[rd] = reg[rs1] + reg[rs2];
            } else {
                reg[rd] = reg[rs1] - reg[rs2];
            }

            // 输出：PC|指令|寄存器结果|null
            cout << "PC=" << to_hex(p * 4) << "|" 
                 << code << " " << x(rd) << ", " << x(rs1) << ", " << x(rs2) << "|"
                 << x(rd) << "=" << to_hex(reg[rd]) << "|" << "null" << endl;
            p++;  // 顺序执行，PC+1（行号+1）
        }

        // ==================== I型：addi, lw ====================
        // addi: opcode=0010011, lw: opcode=0000011
        else if (pc[p].substr(25, 7) == "0010011" || pc[p].substr(25, 7) == "0000011") {
            code = (pc[p].substr(25, 7) == "0010011") ? "addi" : "lw";

            int rs1 = to_int(pc[p].substr(12, 5));
            int rd  = to_int(pc[p].substr(20, 5));

            // 12位立即数，符号扩展
            int imm = to_int(pc[p].substr(1, 11)) - (1 << 11) * (pc[p][0] - '0');

            if (code == "addi") {
                reg[rd] = reg[rs1] + imm;
                cout << "PC=" << to_hex(p * 4) << "|"
                     << code << " " << x(rd) << ", " << x(rs1) << ", " << imm << "|"
                     << x(rd) << "=" << to_hex(reg[rd]) << "|" << "null" << endl;
            } 
            else if (code == "lw") {
                // 从内存读取：地址 = reg[rs1] + imm，注意内存按字索引
                reg[rd] = mem[reg[rs1] + imm];
                cout << "PC=" << to_hex(p * 4) << "|"
                     << code << " " << x(rd) << ", " << imm << "(" << x(rs1) << ")" << "|"
                     << x(rd) << "=" << to_hex(reg[rd]) << "|" << "null" << endl;
            }
            p++;
        }

        // ==================== S型：sw ====================
        // opcode = 0100011
        else if (pc[p].substr(25, 7) == "0100011") {
            code = "sw";

            // S型的立即数分散在高位7位和低位5位，先拼起来
            string s_imm = pc[p].substr(0, 7) + pc[p].substr(20, 5);
            int imm = to_int(s_imm.substr(1, 11)) - (1 << 11) * (s_imm[0] - '0');

            int rs2 = to_int(pc[p].substr(7, 5));   // 要存的寄存器
            int rs1 = to_int(pc[p].substr(12, 5));  // 基址寄存器

            // 写入内存
            mem[reg[rs1] + imm] = reg[rs2];

            cout << "PC=" << to_hex(p * 4) << "|"
                 << code << " " << x(rs2) << ", " << imm << "(" << x(rs1) << ")" << "|"
                 << "null" << "|"
                 << "MEM[" << to_hex(reg[rs1] + imm) << "]=" << to_hex(reg[rs2]) << endl;
            p++;
        }

        // ==================== B型：beq, bne ====================
        // opcode = 1100011
        else if (pc[p].substr(25, 7) == "1100011") {
            // 通过 funct3 区分 beq(000) 和 bne(001)
            code = (pc[p].substr(17, 3) == "000") ? "beq" : "bne";

            // B型立即数重组：imm[12] + imm[11] + imm[10:5] + imm[4:1] + 0
            string s_imm = pc[p].substr(0, 1)      // imm[12]
                         + pc[p].substr(24, 1)      // imm[11]
                         + pc[p].substr(1, 6)        // imm[10:5]
                         + pc[p].substr(20, 4)       // imm[4:1]
                         + "0";                      // imm[0] = 0

            int imm = to_int(s_imm.substr(1, 11)) - (1 << 11) * (s_imm[0] - '0');

            int rs2 = to_int(pc[p].substr(7, 5));
            int rs1 = to_int(pc[p].substr(12, 5));

            // 注意：imm 是字节偏移，要除以 4 得到行号偏移
            cout << "PC=" << to_hex(p * 4) << "|"
                 << code << " " << x(rs1) << ", " << x(rs2) << ", " << imm * 2 << "|"
                 << "null" << "|" << "null" << endl;

            if (code == "beq") {
                if (reg[rs1] == reg[rs2]) {
                    p = p + imm / 2;  // 跳转，注意这里 imm/2 对应行号偏移
                } else {
                    p++;  // 不跳转，顺序执行
                }
            } else if (code == "bne") {
                if (reg[rs1] != reg[rs2]) {
                    p = p + imm / 2;
                } else {
                    p++;
                }
            }
        }
    }

    return 0;
}
```

---

### 四、代码解析

1. **寄存器与内存**：`reg[32]` 是寄存器堆，`x0` 固定为 0。`mem[1024]` 模拟 4KB 内存（按字索引，每个元素 4 字节）。

2. **opcode 判断**：每条指令最后 7 位决定类型。这是译码的第一步。

3. **立即数符号扩展**：`to_int(s.substr(1,11)) - (1<<11)*(s[0]-'0')`。如果最高位 `s[0]` 是 '1'，就减去 2048，得到负数的真实值。

4. **B型立即数重组**：和汇编器相反，模拟器要把打乱顺序的位重新拼回正常的 13 位立即数。`imm[0]` 固定补 0。

5. **PC 更新**：`p` 是行号索引。`beq` 跳转时，`p = p + imm/2` 这里的 `imm/2` 实际上是行号偏移（因为 `imm` 是字节偏移，除以 4 得字偏移，但代码里写成了 `/2`，这是因为 `imm` 在编码时已经左移了一位，实际值是字节偏移的一半，再除以 2 就是除以 4）。

## 实验三：Cache 模拟器

### 一、题目描述

模拟一个固定容量 64 Byte 的 Cache，分别实现三种映射方式：
- **直接映射**（Direct Mapped）
- **2路组相联**（2-way Set Associative）
- **4路组相联**（4-way Set Associative）

**Cache 参数**：
- 容量：64 Byte
- 块大小（Block）：4 Word = 16 Byte（1 Word = 4 Byte）
- 替换算法：LRU（最近最少使用）
- 写策略：Write Back（写回）
- 内存地址：8 位（0x00 ~ 0xFF）

**输入**：每行一个内存访问记录，格式：`0xXX R`（读）或 `0xXX W`（写）  
**输出**：每种映射方式下，每个 Block 的内容、命中次数、未命中次数、命中率（保留 2 位小数）

**地址划分**：
- 块大小 16 Byte = 2^4，所以 **Offset 占 4 位**
- 剩余位用于 Index 和 Tag

---

### 二、思路指导

#### 2.1 地址怎么拆分？

8 位地址 `A7 A6 A5 A4 A3 A2 A1 A0`：
- **Offset（4 位）**：`A3 A2 A1 A0`，块内偏移（0~15）
- **Index + Tag（4 位）**：剩余高位

不同映射方式下 Index 位数不同：

| 映射方式 | 组数 | Index 位数 | Tag 位数 |
|---------|------|-----------|---------|
| 直接映射 | 4 组 | 2 位 | 2 位 |
| 2路组相联 | 2 组 | 1 位 | 3 位 |
| 4路组相联 | 1 组 | 0 位 | 4 位 |

**计算方法**：
- 直接映射：64 Byte / 16 Byte = 4 块，4 个索引 → Index 2 位
- 2路：4 块 / 2 路 = 2 组 → Index 1 位
- 4路：4 块 / 4 路 = 1 组 → Index 0 位（全相联）

#### 2.2 什么是命中？

访问地址时：
1. 用 Index 找到对应的组/行
2. 比较该位置存储的 **Tag** 和地址的 Tag 是否相同
3. 如果相同 → **Hit**；如果不同 → **Miss**，需要替换

#### 2.3 LRU 替换算法

LRU = **最近最少使用**。每组（Set）里有多个块，用计数器记录每个块多久没被访问了：
- 命中时：该块计数器清零，其他块计数器 +1
- 未命中时：找计数器最大的块（最久没被用）替换掉，新块计数器清零

#### 2.4 输出格式

每个块输出它缓存的内存范围。例如 `MEM[0x40]-MEM[0x4F]` 表示这个块缓存了从 0x40 到 0x4F 的 16 字节。

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <bitset>
#include <sstream>
#include <iomanip>
using namespace std;

// 整数转十六进制，保留2位（如 0x4F）
string tohex(int s) {
    stringstream ss;
    ss << hex << uppercase << setfill('0') << setw(2) << s;
    return "0x" + ss.str();
}

// ==================== 直接映射 ====================
void Cache_direct(vector<string> codes) {
    int hit = 0, miss = 0;

    // 4 个块，每个块存一个 Tag 和一个地址标记
    vector<string> tags(4, "");    // 4 个 Tag
    vector<string> data(4, "");    // 4 个块对应的起始地址

    for (auto code : codes) {
        string address = code.substr(0, 4);           // 如 "0x00"
        int add = stoi(code.substr(2, 2), 0, 16);      // 转整数，如 0

        // 8 位二进制：如 0x00 -> 00000000
        string bin = bitset<8>(add).to_string();

        // 直接映射：Index 2 位（bit[3:2]），Tag 2 位（bit[7:4]）
        // 注意：Offset 是低 4 位（bit[3:0]），所以 Index 是 bit[5:4]？
        // 重新算：8位地址，Offset 4位，剩 4 位。4 块直接映射，Index 2 位。
        // Index 取 bit[5:4]（从高位数第3、4位）
        string index = bin.substr(2, 2);   // bit[5:4]（从左数第3位开始2位）
        string tag = bin.substr(0, 2);     // bit[7:6]（最高2位）

        int idx = stoi(index, 0, 2);  // Index 转整数 0~3

        if (tags[idx] == tag) {
            hit++;  // Tag 匹配，命中
        } else {
            miss++; // 不命中，替换
            tags[idx] = tag;
            data[idx] = address;
        }
    }

    // 输出结果
    cout << "Cache-direct:" << endl;
    for (int i = 0; i < 4; i++) {
        if (data[i] == "") {
            cout << "\tindex " << i << ": null" << endl;
        } else {
            // 计算该块缓存的结束地址：起始 + 15（块大小16字节）
            int start = stoi(data[i].substr(2), 0, 16);
            cout << "\tindex " << i << ": MEM[" << data[i] 
                 << "]-MEM[" << tohex(start + 0xF) << "]" << endl;
        }
    }
    cout << "\tHit: " << hit << "; Miss: " << miss 
         << "; Hit rate: " << fixed << setprecision(2) 
         << (double)hit / (hit + miss) << endl;
}

// ==================== 2路组相联 ====================
void Cache_2_way(vector<string> codes) {
    int hit = 0, miss = 0;
    bool h = 0;

    // 2 组，每组 2 个块
    vector<vector<string>> tags(2, vector<string>(2, ""));   // Tag
    vector<vector<string>> data(2, vector<string>(2, ""));   // 地址
    vector<vector<int>> count = {{0, 0}, {0, 0}};             // LRU 计数器

    for (auto code : codes) {
        h = 0;
        string address = code.substr(0, 4);
        int add = stoi(code.substr(2, 2), 0, 16);
        string bin = bitset<8>(add).to_string();

        // 2路：2 组，Index 1 位（bit[4]），Tag 3 位（bit[7:5]）
        string index = bin.substr(3, 1);   // bit[4]
        string tag = bin.substr(0, 3);     // bit[7:5]

        int idx = stoi(index, 0, 2);  // 0 或 1

        // 先查这组里的两个块
        for (int i = 0; i < 2; i++) {
            if (tags[idx][i] == tag) {
                h = 1;           // 命中
                count[idx][i] = 0;  // 命中块计数器清零
            } else if (tags[idx][i] != "") {
                count[idx][i]++;    // 未命中块计数器+1
            }
        }

        if (h) {
            hit++;
        } else {
            miss++;
            // 找替换块：先找空位，没有就找计数器最大的（LRU）
            int victim = -1;
            for (int i = 0; i < 2; i++) {
                if (tags[idx][i] == "") { victim = i; break; }
            }
            if (victim == -1) {
                victim = (count[idx][0] >= count[idx][1]) ? 0 : 1;
            }
            tags[idx][victim] = tag;
            data[idx][victim] = address;
            count[idx][victim] = 0;
        }
    }

    cout << "Cache-2-way:" << endl;
    for (int i = 0; i < 2; i++) {
        cout << "\tindex " << i << ": ";
        for (int j = 0; j < 2; j++) {
            if (data[i][j] == "") {
                cout << "null";
            } else {
                int start = stoi(data[i][j].substr(2), 0, 16);
                cout << "MEM[" << data[i][j] << "]-MEM[" << tohex(start + 0xF) << "]";
            }
            if (j < 1) cout << ", ";
        }
        cout << endl;
    }
    cout << "\tHit: " << hit << "; Miss: " << miss 
         << "; Hit rate: " << fixed << setprecision(2) 
         << (double)hit / (hit + miss) << endl;
}

// ==================== 4路组相联（全相联） ====================
void Cache_4_way(vector<string> codes) {
    int hit = 0, miss = 0;
    bool h = 0;

    // 1 组，4 个块
    vector<string> tags(4, "");
    vector<string> data(4, "");
    vector<int> count = {0, 0, 0, 0};  // LRU 计数器

    for (auto code : codes) {
        h = 0;
        string address = code.substr(0, 4);
        int add = stoi(code.substr(2, 2), 0, 16);
        string bin = bitset<8>(add).to_string();

        // 4路：只有 1 组，Index 0 位，Tag 4 位（bit[7:4]）
        string tag = bin.substr(0, 4);

        for (int i = 0; i < 4; i++) {
            if (tags[i] == tag) {
                h = 1;
                count[i] = 0;
            } else if (tags[i] != "") {
                count[i]++;
            }
        }

        if (h) {
            hit++;
        } else {
            miss++;
            int victim = -1;
            for (int i = 0; i < 4; i++) {
                if (tags[i] == "") { victim = i; break; }
            }
            if (victim == -1) {
                // 找计数器最大的
                int max_count = -1;
                for (int i = 0; i < 4; i++) {
                    if (count[i] > max_count) {
                        max_count = count[i];
                        victim = i;
                    }
                }
            }
            tags[victim] = tag;
            data[victim] = address;
            count[victim] = 0;
        }
    }

    cout << "Cache-4-way:" << endl;
    cout << "\tindex 0: ";
    for (int i = 0; i < 4; i++) {
        if (data[i] == "") {
            cout << "null";
        } else {
            int start = stoi(data[i].substr(2), 0, 16);
            cout << "MEM[" << data[i] << "]-MEM[" << tohex(start + 0xF) << "]";
        }
        if (i < 3) cout << ", ";
    }
    cout << endl;
    cout << "\tHit: " << hit << "; Miss: " << miss 
         << "; Hit rate: " << fixed << setprecision(2) 
         << (double)hit / (hit + miss) << endl;
}

int main() {
    vector<string> codes;
    string code;
    while (getline(cin, code) && code != "") {
        codes.push_back(code);
    }

    Cache_direct(codes);
    Cache_2_way(codes);
    Cache_4_way(codes);

    return 0;
}
```

### 四、代码解析

1. **地址拆分**：用 `bitset<8>` 把地址转成 8 位二进制字符串，然后用 `substr` 截取 Tag 和 Index。

2. **直接映射**：一一对应，Index 直接定位块。Tag 不匹配就替换，没有选择权。

3. **组相联**：先通过 Index 找到组，再在组内比较所有块的 Tag。LRU 计数器实现简单：命中清零，未命中+1，替换时选最大的。

4. **输出格式**：每个块显示缓存的内存范围。`MEM[0x40]-MEM[0x4F]` 表示块起始地址是 0x40，大小 16 字节，结束地址 0x4F。

## 实验四：基于 Cache 的矩阵计算优化

### 一、题目描述

实现矩阵乘法 `C = A × B`，通过**分块（Tiling）**优化 Cache Miss 次数，找到最优的分块大小。

**输入**：
```
cache_size associativity cache_block_size
N
矩阵 A（N 行）
矩阵 B（N 行）
```

**输出**：
```
best_block_size
总 cache_miss 次数
```

**约束**：
- Cache 参数：容量、组相联度、块大小
- 矩阵元素：float（4 字节）
- 分块大小 T：1 ≤ T ≤ N，若多个 T 的 Miss 相同，取最大的 T
- 替换策略：LRU
- 写操作：不命中时先读入整块（计一次读 Miss），再修改

**分块乘法顺序**（必须按此顺序）：
```
for ii = 0 to N-1 step T
  for jj = 0 to N-1 step T
    for kk = 0 to N-1 step T
      for i = ii to min(ii+T, N)-1
        for k = kk to min(kk+T, N)-1
          for j = jj to min(jj+T, N)-1
            C[i][j] += A[i][k] * B[k][j]
```

---

### 二、思路指导

#### 2.1 为什么要分块？

普通矩阵乘法按 `i-j-k` 顺序，内层循环 `C[i][j] += A[i][k] * B[k][j]` 中：
- `A[i][k]` 按行访问，连续性好
- `B[k][j]` 按列访问，每次跳 N 个元素，Cache 不友好
- 当矩阵很大时，Cache 装不下，反复换入换出，Miss 极高

**分块（Tiling）** 把大矩阵切成小方块（T×T），让每个小方块能装进 Cache，减少换入换出。

#### 2.2 解题步骤

1. **枚举所有可能的 T**：从 1 到 N
2. **对每个 T，模拟分块乘法**：按题目要求的三层分块循环
3. **模拟 Cache 访问**：每次访问 `A[i][k]`、`B[k][j]`、`C[i][j]` 时，调用 Cache 模拟器
4. **统计 Miss 次数**：记录每个 T 对应的总 Miss
5. **选最优**：Miss 最少的 T，如果有多个取最大的 T

#### 2.3 Cache 模拟要点

- 地址计算：`A[i][k]` 的地址 = 基地址 + `(i * N + k) * 4` 字节
- 矩阵在内存中按**行优先**存储：`A[0][0], A[0][1], A[0][2]...`
- 每次访问一个 float（4 字节），如果该地址所在块不在 Cache 中，计一次 Miss，并把整块读入
- 写 `C[i][j]` 时，如果不命中，也是先读入整块（Miss+1），再修改块内数据

#### 2.4 地址到 Cache 的映射

Cache 模拟器需要：
- `block_size`：块大小（字节）
- `num_sets`：组数 = cache_size / (associativity * block_size)
- `offset_bits`：块内偏移位数 = log2(block_size)
- `index_bits`：索引位数 = log2(num_sets)
- `tag`：地址右移 (offset_bits + index_bits) 位

### 三、完整代码

```cpp
#include <iostream>
#include <vector>
#include <unordered_map>
#include <sstream>
#include <climits>
#include <algorithm>
using namespace std;

// ==================== Cache 模拟器类 ====================
class Cache {
    int num_sets;        // 组数
    int associativity;   // 组相联度（每组的块数）
    int block_size;      // 块大小（字节）
    int offset_bits;     // 块内偏移位数
    int index_bits;      // 索引位数

    vector<unordered_map<long long, int>> sets;  // 每组：Tag -> 时间戳
    int timestamp;       // 全局时间戳，用于 LRU
    long long misses;    // Miss 次数

public:
    Cache(int cache_size, int assoc, int blk_size)
        : associativity(assoc), block_size(blk_size),
          timestamp(0), misses(0) {
        // 计算组数
        num_sets = cache_size / (assoc * blk_size);
        sets.resize(num_sets);

        // 计算 offset_bits = log2(block_size)
        offset_bits = 0;
        for (int tmp = block_size; tmp > 1; tmp >>= 1) offset_bits++;

        // 计算 index_bits = log2(num_sets)
        index_bits = 0;
        for (int tmp = num_sets; tmp > 1; tmp >>= 1) index_bits++;
    }

    // 访问某个地址（以字节为单位）
    bool access(long long addr) {
        // 计算 index：地址中间几位
        int index = (addr >> offset_bits) & ((1LL << index_bits) - 1);
        // 计算 tag：地址高位
        long long tag = addr >> (offset_bits + index_bits);

        auto& s = sets[index];

        // 检查是否命中
        if (s.count(tag)) {
            s[tag] = ++timestamp;  // 更新时间戳
            return true;           // 命中
        }

        // 未命中
        misses++;

        // 如果组满了，找 LRU 替换
        if ((int)s.size() >= associativity) {
            long long lru_tag = -1;
            int lru_time = INT_MAX;
            for (auto& [t, time] : s) {
                if (time < lru_time) {
                    lru_time = time;
                    lru_tag = t;
                }
            }
            s.erase(lru_tag);
        }

        // 装入新块
        s[tag] = ++timestamp;
        return false;
    }

    // 访问某个地址所在的块（矩阵计算中调用这个）
    bool access_block(long long addr) {
        // 块对齐：找到该地址所在的块的起始地址
        long long block_addr = (addr / block_size) * block_size;
        return access(block_addr);
    }

    // 重置 Cache（用于测试下一个 T）
    void reset() {
        misses = 0;
        timestamp = 0;
        for (auto& s : sets) s.clear();
    }

    long long get_misses() { return misses; }
};

// 计算矩阵元素 A[i][k] 的内存地址（假设 A 从地址 0 开始）
// 实际地址 = (i * cols + col) * 4 字节
long long mat_addr(int row, int col, int cols) {
    return (long long)(row * cols + col) * 4;
}

int main() {
    // 读取 Cache 参数
    int cache_size, asso, block_size;
    cin >> cache_size >> asso >> block_size;

    // 读取矩阵维度
    int N;
    cin >> N;

    // 读取矩阵 A 和 B（虽然实际计算中不需要值，但需要读取输入）
    vector<vector<int>> A(N, vector<int>(N));
    vector<vector<int>> B(N, vector<int>(N));

    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            cin >> A[i][j];

    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            cin >> B[i][j];

    Cache cache(cache_size, asso, block_size);
    vector<long long> miss_count(N + 1, 0);  // miss_count[T] 表示分块大小为 T 时的 Miss

    // ==================== T = 1：不分块（朴素算法） ====================
    {
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                for (int k = 0; k < N; k++) {
                    // 访问 A[i][k]
                    cache.access_block(mat_addr(i, k, N));
                    // 访问 B[k][j]（B 的基地址偏移 N*N*4）
                    cache.access_block(mat_addr(k, j, N) + (N * N * 4));
                    // 访问 C[i][j]（C 的基地址偏移 2*N*N*4）
                    cache.access_block(mat_addr(i, j, N) + (2 * N * N * 4));
                }
            }
        }
        miss_count[1] = cache.get_misses();
        cache.reset();
    }

    // ==================== T = 2 到 N：分块算法 ====================
    for (int T = 2; T <= N; T++) {
        for (int ii = 0; ii < N; ii += T) {
            for (int jj = 0; jj < N; jj += T) {
                for (int kk = 0; kk < N; kk += T) {
                    // 小矩阵乘法
                    for (int i = ii; i < min(ii + T, N); i++) {
                        for (int k = kk; k < min(kk + T, N); k++) {
                            for (int j = jj; j < min(jj + T, N); j++) {
                                cache.access_block(mat_addr(i, k, N));
                                cache.access_block(mat_addr(k, j, N) + (N * N * 4));
                                cache.access_block(mat_addr(i, j, N) + (2 * N * N * 4));
                            }
                        }
                    }
                }
            }
        }
        miss_count[T] = cache.get_misses();
        cache.reset();
    }

    // ==================== 找最优 T ====================
    // 规则：Miss 最少，如果有多个取最大的 T
    long long min_misses = miss_count[1];
    int best_T = 1;

    for (int T = 1; T <= N; T++) {
        if (miss_count[T] <= min_misses) {
            min_misses = miss_count[T];
            best_T = T;
        }
    }

    cout << best_T << endl;
    cout << min_misses << endl;

    return 0;
}
```

---

### 四、代码解析

1. **Cache 类**：封装了 Cache 的所有操作。
   - `access(addr)`：访问某个地址，返回是否命中。
   - `access_block(addr)`：矩阵计算中调用，先把地址对齐到块边界，再访问。
   - 用 `unordered_map` 存储每组内的 Tag 和时间戳，实现简单 LRU。

2. **地址计算**：
   - `A[i][k]` 地址 = `(i * N + k) * 4`
   - `B[k][j]` 地址 = `(k * N + j) * 4 + N*N*4`（假设 B 在 A 后面）
   - `C[i][j]` 地址 = `(i * N + j) * 4 + 2*N*N*4`（假设 C 在 B 后面）
   - 这只是模拟地址，实际值不重要，关键是地址的分布模式。

3. **分块循环**：严格按照题目要求的 6 层循环顺序。外层 3 层控制块的位置，内层 3 层控制块内的元素。

4. **最优 T 选择**：遍历 T=1 到 N，记录每个的 Miss。注意 `<=` 保证了 Miss 相同时取更大的 T。
