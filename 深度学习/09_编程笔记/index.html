# 手动实现带正则化的梯度下降，理解权重衰减
# ============================================
class RegularizedLinearRegression:
    """
    带 L1/L2 正则化的线性回归
    展示正则化如何影响参数更新
    """

    def __init__(self, l1_reg=0, l2_reg=0):
        self.l1_reg = l1_reg
        self.l2_reg = l2_reg
        self.w = None

    def fit(self, X, y, lr=0.01, epochs=100):
        n_features = X.shape[1]
        self.w = np.zeros(n_features)

        history = {'w': [], 'loss': []}

        for _ in range(epochs):
            # 预测
            y_pred = X @ self.w

            # MSE 梯度
            grad = 2 * X.T @ (y_pred - y) / len(y)

            # L2 梯度: λ * w
            if self.l2_reg > 0:
                grad_l2 = self.l2_reg * self.w
            else:
                grad_l2 = 0

            # L1 梯度: λ * sign(w)，次梯度
            if self.l1_reg > 0:
                grad_l1 = self.l1_reg * np.sign(self.w)
                grad_l1[np.abs(self.w) < 1e-6] = 0  # 阈值处理
            else:
                grad_l1 = 0

            # 总梯度
            total_grad = grad + grad_l2 + grad_l1

            # 更新
            self.w -= lr * total_grad

            # 记录
            history['w'].append(self.w.copy())
            history['loss'].append(np.mean((y_pred - y)**2))

        return history

# Dropout 实现
# ============================================
class DropoutLayer:
    """
    Dropout 层

    训练时: 以概率 p 保留神经元，其余置零
    测试时: 所有神经元参与，但输出乘以 p（保持期望一致）
    """

    def __init__(self, keep_prob=0.5):
        self.keep_prob = keep_prob
        self.mask = None  # 保存训练时的 mask

    def forward(self, x, training=True):
        if training:
            # 随机生成 mask
            self.mask = np.random.rand(*x.shape) < self.keep_prob
            return x * self.mask / self.keep_prob  # 缩放保持期望
        else:
            return x  # 测试时不 dropout

    def backward(self, grad):
        # 反向传播时，梯度也只通过保留的神经元
        return grad * self.mask / self.keep_prob

# ============================================
# 完整的带 Dropout 的神经网络
# ============================================
class NeuralNetWithDropout:
    """
    带 Dropout 的浅层神经网络
    """

    def __init__(self, input_dim, hidden_dim, output_dim, keep_prob=0.5):
        self.keep_prob = keep_prob

        # 初始化权重（He初始化，适合ReLU）
        self.W1 = np.random.randn(hidden_dim, input_dim) * np.sqrt(2/input_dim)
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(output_dim, hidden_dim) * np.sqrt(2/hidden_dim)
        self.b2 = np.zeros(output_dim)

    def relu(self, x):
        return np.maximum(0, x)

    def relu_grad(self, x):
        return (x > 0).astype(float)

    def forward(self, X, training=True):
        # 第一层
        self.z1 = X @ self.W1.T + self.b1
        self.h1 = self.relu(self.z1)

        # Dropout
        if training:
            self.drop_mask = np.random.rand(*self.h1.shape) < self.keep_prob
            self.h1_drop = self.h1 * self.drop_mask / self.keep_prob
        else:
            self.h1_drop = self.h1

        # 输出层
        self.z2 = self.h1_drop @ self.W2.T + self.b2

        return self.z2

    def backward(self, y, y_pred):
        m = len(y)

        # 输出层梯度
        dL_dz2 = (y_pred - y).reshape(-1, 1)
        dL_dW2 = dL_dz2.T @ self.h1_drop / m
        dL_db2 = np.mean(dL_dz2)

        # 隐藏层梯度
        dL_dh1 = dL_dz2 @ self.W2
        dL_dz1 = dL_dh1 * self.relu_grad(self.z1) * self.drop_mask / self.keep_prob
        dL_dW1 = dL_dz1.T @ y / m
        dL_db1 = np.mean(dL_dz1)

        return {'W1': dL_dW1, 'b1': dL_db1, 'W2': dL_dW2, 'b2': dL_db2}

# 对比：有 Dropout vs 无 Dropout
# ============================================
np.random.seed(0)

# 生成数据（容易过拟合）
N = 100
x_train = np.random.randn(N, 1)
y_train = np.sin(3*x_train.flatten()) * np.exp(-x_train.flatten()**2/4) + np.random.normal(0, 0.1, N)

# 网络参数
input_dim = 1
hidden_dim = 100  # 容易过拟合的大网络
output_dim = 1

# 训练函数
def train_network(net, x, y, epochs=1000, lr=0.01, verbose=True):
    losses = []
    for epoch in range(epochs):
        # 前向
        y_pred = net.forward(x, training=True)
        loss = np.mean((y_pred.flatten() - y)**2)
        losses.append(loss)

        # 反向
        grads = net.backward(y, y_pred)

        # 更新
        net.W1 -= lr * grads['W1']
        net.b1 -= lr * grads['b1']
        net.W2 -= lr * grads['W2']
        net.b2 -= lr * grads['b2']

        if verbose and epoch % 200 == 0:
            print(f"Epoch {epoch}: loss = {loss:.6f}")

    return losses

# 无 Dropout

# 早停实现
# ============================================
class EarlyStopping:
    """
    早停策略

    监控验证集损失，连续 patience 轮不改善则停止
    """

    def __init__(self, patience=10, min_delta=0.0001):
        self.patience = patience
        self.min_delta = min_delta
        self.best_loss = float('inf')
        self.counter = 0
        self.should_stop = False
        self.best_params = None

    def check(self, val_loss, model):
        if val_loss < self.best_loss - self.min_delta:
            # 有改善
            self.best_loss = val_loss
            self.counter = 0
            self.best_params = {k: v.copy() for k, v in model.params.items()}
        else:
            # 没有改善
            self.counter += 1
            if self.counter >= self.patience:
                self.should_stop = True
                return True
        return False

# 简化版神经网络（用于早停演示）
class SimpleNet:
    def __init__(self, layer_dims):
        self.params = {}
        for i in range(1, len(layer_dims)):
            self.params[f'W{i}'] = np.random.randn(layer_dims[i], layer_dims[i-1]) * 0.1
            self.params[f'b{i}'] = np.zeros(layer_dims[i])

    def relu(self, x):
        return np.maximum(0, x)

    def forward(self, X):
        A = X
        for i in range(1, len(self.params)//2 + 1):
            Z = A @ self.params[f'W{i}'].T + self.params[f'b{i}']
            A = self.relu(Z)
        return A

# 早停 vs 训练到底
# ============================================
# 生成数据

x_train = np.random.randn(N, 1)
y_train = np.sin(3*x_train.flatten()) * np.exp(-x_train.flatten()**2/4) + np.random.normal(0, 0.1, N)

# 划分验证集

np.random.shuffle(indices)
train_idx = indices[:70]
val_idx = indices[70:]

x_tr, y_tr = x_train[train_idx], y_train[train_idx]
x_val, y_val = x_train[val_idx], y_train[val_idx]

# 训练网络（训练到底）
net_full = SimpleNet([1, 64, 1])
train_losses_full, val_losses_full = [], []

for epoch in range(500):
    # 前向
    y_pred = net_full.forward(x_tr)
    train_loss = np.mean((y_pred.flatten() - y_tr)**2)
    train_losses_full.append(train_loss)

    # 反向（简化）
    grad = 2 * (y_pred.flatten() - y_tr).reshape(-1,1) / len(y_tr)
    net_full.params['W1'] -= 0.01 * (grad @ x_tr)
    net_full.params['W2'] -= 0.01 * grad.T

    # 验证
    y_val_pred = net_full.forward(x_val)
    val_loss = np.mean((y_val_pred.flatten() - y_val)**2)
    val_losses_full.append(val_loss)

# 训练网络（早停）
net_early = SimpleNet([1, 64, 1])
train_losses_early, val_losses_early = [], []
early_stop = EarlyStopping(patience=20)
best_epoch = 0

for epoch in range(500):
    y_pred = net_early.forward(x_tr)
    train_loss = np.mean((y_pred.flatten() - y_tr)**2)
    train_losses_early.append(train_loss)

    grad = 2 * (y_pred.flatten() - y_tr).reshape(-1,1) / len(y_tr)
    net_early.params['W1'] -= 0.01 * (grad @ x_tr)
    net_early.params['W2'] -= 0.01 * grad.T

    y_val_pred = net_early.forward(x_val)
    val_loss = np.mean((y_val_pred.flatten() - y_val)**2)
    val_losses_early.append(val_loss)

    if early_stop.check(val_loss, net_early):
        best_epoch = epoch - early_stop.patience
        break

# 画图
plt.figure(figsize=(10, 4))

plt.subplot(1, 2, 1)
plt.plot(train_losses_full, label='训练集', alpha=0.7)
plt.plot(val_losses_full, label='验证集', alpha=0.7)
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('训练到底：验证集损失反弹')
plt.legend()
plt.yscale('log')

plt.subplot(1, 2, 2)
plt.plot(train_losses_early, label='训练集', alpha=0.7)
plt.plot(val_losses_early, label='验证集', alpha=0.7)
if best_epoch > 0:
    plt.axvline(best_epoch, color='red', linestyle='--', label=f'早停 epoch={best_epoch}')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('早停：在验证集最优点停止')
plt.legend()
plt.yscale('log')

plt.tight_layout()
plt.show()

print(f"\n早停效果:")
print(f"  最佳验证损失 epoch: {best_epoch}")
print(f"  训练到底时最终验证损失: {val_losses_full[-1]:.6f}")
print(f"  早停时验证损失: {val_losses_early[best_epoch]:.6f}")

    五、关键知识点回顾¶

    正则化方法 原理 代码实现 适用场景
    L2 (Ridge) λ‖w‖² 惩罚 梯度 + λw 几乎所有场景
    L1 (Lasso) λ‖w‖₁ 惩罚 梯度 + λ·sign(w) 特征选择
    Dropout 随机丢弃神经元 mask / keep_prob 大型网络
    早停 监控验证集 patience 轮不改善停止 任何训练
    数据增强 增加训练样本 - 图像/语音

    六、思考题¶

    1. Dropout 为什么让测试时乘以 keep_prob？ 提示：保持期望一致。

    2. L1 和 L2 哪个正则化更强？ 试试不同 λ 值观察效果。

    3. 正则化和学习率衰减有什么联系？ 提示：梯度下降的隐式正则化效应。
