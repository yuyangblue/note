    import numpy as np
import matplotlib.pyplot as plt

# 生成一个简单的优化问题
np.random.seed(42)

def generate_data(n=100):
    """生成回归数据 y = 3x + 5 + noise"""
    x = np.random.uniform(-5, 5, n)
    y = 3 * x + 5 + np.random.normal(0, 2, n)
    return x, y

x_data, y_data = generate_data(100)

# 定义线性模型
class LinearModel:
    def __init__(self):
        self.w = 0.0
        self.b = 0.0

    def predict(self, x):
        return self.w * x + self.b

    def loss(self, x, y):
        return np.mean((self.predict(x) - y) ** 2)

    def grad(self, x, y):
        """解析梯度"""
        pred = self.predict(x)
        d_w = 2 * np.mean((pred - y) * x)
        d_b = 2 * np.mean(pred - y)
        return d_w, d_b

# ============================================
# 梯度下降变体对比
# ============================================
def train(model, x, y, lr=0.01, epochs=100, batch_size=None, verbose=True):
    """
    训练模型

    batch_size:
        None 或 len(x): 全批量 GD
        1: 随机梯度下降 (SGD)
        k (1 < k < n): Mini-batch GD
    """
    n = len(x)
    losses = []
    w_history = [model.w]

    for epoch in range(epochs):
        if batch_size is None or batch_size >= n:
            # 全批量
            d_w, d_b = model.grad(x, y)
        else:
            # Mini-batch：随机选一批
            idx = np.random.choice(n, batch_size, replace=False)
            d_w, d_b = model.grad(x[idx], y[idx])

        # 更新
        model.w -= lr * d_w
        model.b -= lr * d_b

        losses.append(model.loss(x, y))
        w_history.append(model.w)

    return np.array(losses), np.array(w_history)

# 对比不同的 batch_size
lr = 0.001
epochs = 50

# 全批量 GD
model_full = LinearModel()
loss_full, _ = train(model_full, x_data, y_data, lr=lr, epochs=epochs, batch_size=len(x_data))

# Mini-batch SGD (batch=10)
model_sgd = LinearModel()
loss_sgd, _ = train(model_sgd, x_data, y_data, lr=lr, epochs=epochs, batch_size=10)

# 随机 SGD (batch=1)
model_sgd1 = LinearModel()
loss_sgd1, _ = train(model_sgd1, x_data, y_data, lr=lr*0.1, epochs=epochs, batch_size=1)

# 画对比图
plt.figure(figsize=(12, 4))

plt.subplot(1, 3, 1)
plt.plot(loss_full, label='Full GD')
plt.plot(loss_sgd, label='Mini-batch (B=10)')
plt.plot(loss_sgd1, label='SGD (B=1)')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('损失曲线对比')
plt.legend()
plt.yscale('log')
plt.grid(True, alpha=0.3)

# 收敛速度对比
plt.subplot(1, 3, 2)
plt.bar(['Full GD', 'Mini-batch', 'SGD'], 
        [loss_full[-1], loss_sgd[-1], loss_sgd1[-1]])
plt.ylabel('最终损失')
plt.title('收敛到最优解')
plt.grid(True, alpha=0.3)

# 噪声效果可视化
plt.subplot(1, 3, 3)
plt.plot(loss_full, label='Full GD (平滑)', linewidth=2)
plt.plot(loss_sgd, label='Mini-batch (有噪声)', alpha=0.7)
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('梯度噪声的影响')
plt.legend()
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

print(f"最终结果对比:")
print(f"  全批量 GD:  w={model_full.w:.3f}, b={model_full.b:.3f}, loss={loss_full[-1]:.4f}")
print(f"  Mini-batch:  w={model_sgd.w:.3f}, b={model_sgd.b:.3f}, loss={loss_sgd[-1]:.4f}")
print(f"  SGD:         w={model_sgd1.w:.3f}, b={model_sgd1.b:.3f}, loss={loss_sgd1[-1]:.4f}")
print(f"\n真实参数: w=3.000, b=5.000")

# Momentum 优化器实现
# ============================================
class MomentumOptimizer:
    """
    带动量的梯度下降

    核心思想：
    - 速度 v 累积历史梯度
    - 参数更新 = 速度 v（而不是单次梯度）
    - 这样可以冲出局部平坦区域，沿主要方向加速
    """

    def __init__(self, lr=0.01, momentum=0.9):
        self.lr = lr
        self.momentum = momentum
        self.v = None  # 速度

    def step(self, params, grads):
        """
        params: 参数字典 {'w': value, 'b': value, ...}
        grads: 梯度字典 {'w': grad, 'b': grad, ...}
        """
        if self.v is None:
            # 初始化速度为0
            self.v = {k: np.zeros_like(v) for k, v in params.items()}

        for key in params:
            # v_t = β * v_{t-1} + g_t
            self.v[key] = self.momentum * self.v[key] + grads[key]
            # θ_t = θ_{t-1} - α * v_t
            params[key] -= self.lr * self.v[key]

# ============================================
# 可视化 Momentum 的效果
# ============================================
def visualize_momentum():
    """
    在一个椭圆山谷上比较 GD 和 Momentum
    """
    # 定义损失函数：f(w, b) = w² + 10*b²
    # 特点：一个方向陡峭(w)，一个方向平缓(b)
    def loss_func(w, b):
        return w**2 + 10 * b**2

    def grad_func(w, b):
        return np.array([2*w, 20*b])

    # GD
    lr = 0.1
    w_gd, b_gd = 1.0, 1.0
    path_gd = [(w_gd, b_gd)]

    for _ in range(30):
        dw, db = grad_func(w_gd, b_gd)
        w_gd -= lr * dw
        b_gd -= lr * db
        path_gd.append((w_gd, b_gd))

    # Momentum
    lr = 0.1
    momentum = 0.9
    w_mom, b_mom = 1.0, 1.0
    v_w, v_b = 0.0, 0.0
    path_mom = [(w_mom, b_mom)]

    for _ in range(30):
        dw, db = grad_func(w_mom, b_mom)
        v_w = momentum * v_w + dw
        v_b = momentum * v_b + db
        w_mom -= lr * v_w
        b_mom -= lr * v_b
        path_mom.append((w_mom, b_mom))

    # 画图
    plt.figure(figsize=(10, 5))

    # 等高线
    w_range = np.linspace(-1.2, 1.2, 100)
    b_range = np.linspace(-1.2, 1.2, 100)
    W, B = np.meshgrid(w_range, b_range)
    Z = W**2 + 10*B**2

    plt.contour(W, B, Z, levels=np.logspace(-2, 2, 20), cmap='viridis', alpha=0.5)

    # 路径
    path_gd = np.array(path_gd)
    path_mom = np.array(path_mom)

    plt.plot(path_gd[:, 0], path_gd[:, 1], 'b-o', label='GD', markersize=4)
    plt.plot(path_mom[:, 0], path_mom[:, 1], 'r-s', label='Momentum', markersize=4)

    plt.plot(0, 0, '*', color='green', markersize=15, label='最优解')
    plt.xlabel('w')
    plt.ylabel('b')
    plt.title('GD vs Momentum：椭圆山谷')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.show()

visualize_momentum()
print("\n观察：Momentum 在平缓方向累积速度，比 GD 更快到达中心！")

    # ============================================
# Adam 优化器实现
# ============================================
class Adam:
    """
    Adam (Adaptive Moment Estimation)

    结合了：
    - Momentum（一阶动量）：方向
    - RMSprop（二阶动量）：步长

    核心公式：
    m_t = β₁*m_{t-1} + (1-β₁)*g_t      # 一阶动量（均值）
    v_t = β₂*v_{t-1} + (1-β₂)*g_t²     # 二阶动量（方差）

    偏差校正：
    m̂_t = m_t / (1 - β₁^t)
    v̂_t = v_t / (1 - β₂^t)

    更新：θ_t = θ_{t-1} - α * m̂_t / (√v̂_t + ε)
    """

    def __init__(self, lr=0.001, beta1=0.9, beta2=0.999, eps=1e-8):
        self.lr = lr
        self.beta1 = beta1
        self.beta2 = beta2
        self.eps = eps
        self.m = None  # 一阶动量
        self.v = None  # 二阶动量
        self.t = 0  # 时间步

    def step(self, params, grads):
        self.t += 1

        if self.m is None:
            self.m = {k: np.zeros_like(v) for k, v in params.items()}
            self.v = {k: np.zeros_like(v) for k, v in params.items()}

        for key in params:
            # 一阶动量
            self.m[key] = self.beta1 * self.m[key] + (1 - self.beta1) * grads[key]
            # 二阶动量
            self.v[key] = self.beta2 * self.v[key] + (1 - self.beta2) * (grads[key] ** 2)

            # 偏差校正
            m_hat = self.m[key] / (1 - self.beta1 ** self.t)
            v_hat = self.v[key] / (1 - self.beta2 ** self.t)

            # 更新
            params[key] -= self.lr * m_hat / (np.sqrt(v_hat) + self.eps)

# ============================================
# 优化器大比拼
# ============================================
def compare_optimizers():
    """
    在一个复杂曲面上比较不同优化器
    """
    np.random.seed(42)

    # 生成有局部最优的损失曲面
    def rosenbrock(x, y):
        """Rosenbrock 函数：经典的测试函数"""
        return (1 - x)**2 + 100*(y - x**2)**2

    def grad_rosenbrock(x, y):
        dx = -2*(1 - x) - 400*x*(y - x**2)
        dy = 200*(y - x**2)
        return np.array([dx, dy])

    # 优化器配置
    optimizers = {
        'SGD (lr=0.001)': {'type': 'sgd', 'lr': 0.001},
        'Momentum (lr=0.001)': {'type': 'momentum', 'lr': 0.001, 'beta': 0.9},
        'RMSprop (lr=0.01)': {'type': 'rmsprop', 'lr': 0.01, 'beta': 0.9},
        'Adam (lr=0.01)': {'type': 'adam', 'lr': 0.01},
    }

    results = {}

    for name, config in optimizers.items():
        x, y = -2.0, -1.0  # 初始位置
        path = [(x, y)]

        if config['type'] == 'sgd':
            lr = config['lr']
            for _ in range(200):
                dx, dy = grad_rosenbrock(x, y)
                x -= lr * dx
                y -= lr * dy
                path.append((x, y))

        elif config['type'] == 'momentum':
            lr, beta = config['lr'], config['beta']
            v_x, v_y = 0.0, 0.0
            for _ in range(200):
                dx, dy = grad_rosenbrock(x, y)
                v_x = beta * v_x + dx
                v_y = beta * v_y + dy
                x -= lr * v_x
                y -= lr * v_y
                path.append((x, y))

        elif config['type'] == 'rmsprop':
            lr, beta = config['lr'], config['beta']
            eps = 1e-8
            g_sq_x, g_sq_y = 0.0, 0.0
            for _ in range(200):
                dx, dy = grad_rosenbrock(x, y)
                g_sq_x = beta * g_sq_x + (1-beta) * dx**2
                g_sq_y = beta * g_sq_y + (1-beta) * dy**2
                x -= lr * dx / (np.sqrt(g_sq_x) + eps)
                y -= lr * dy / (np.sqrt(g_sq_y) + eps)
                path.append((x, y))

        elif config['type'] == 'adam':
            lr, beta1, beta2 = config['lr'], 0.9, 0.999
            eps = 1e-8
            m_x, m_y = 0.0, 0.0
            v_x, v_y = 0.0, 0.0
            for t in range(1, 201):
                dx, dy = grad_rosenbrock(x, y)
                m_x = beta1 * m_x + (1-beta1) * dx
                m_y = beta1 * m_y + (1-beta1) * dy
                v_x = beta2 * v_x + (1-beta2) * dx**2
                v_y = beta2 * v_y + (1-beta2) * dy**2
                m_x_hat = m_x / (1 - beta1**t)
                m_y_hat = m_y / (1 - beta1**t)
                v_x_hat = v_x / (1 - beta2**t)
                v_y_hat = v_y / (1 - beta2**t)
                x -= lr * m_x_hat / (np.sqrt(v_x_hat) + eps)
                y -= lr * m_y_hat / (np.sqrt(v_y_hat) + eps)
                path.append((x, y))

        results[name] = {
            'path': np.array(path),
            'final_loss': rosenbrock(x, y)
        }

    # 画图
    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    # 等高线
    x_range = np.linspace(-2.5, 2.5, 200)
    y_range = np.linspace(-2.5, 2.5, 200)
    X, Y = np.meshgrid(x_range, y_range)
    Z = rosenbrock(X, Y)

    axes[0].contour(X, Y, Z, levels=np.logspace(-1, 4, 30), cmap='viridis', alpha=0.5)
    colors = ['blue', 'orange', 'green', 'red']
    for i, (name, res) in enumerate(results.items()):
        axes[0].plot(res['path'][:, 0], res['path'][:, 1], 
                    color=colors[i], label=name, linewidth=1.5, alpha=0.7)
    axes[0].plot(1, 1, '*', color='white', markersize=15, markeredgecolor='black')
    axes[0].set_xlabel('x')
    axes[0].set_ylabel('y')
    axes[0].set_title('Rosenbrock 函数上的优化轨迹')
    axes[0].legend(loc='upper left')

    # 损失曲线
    for i, (name, res) in enumerate(results.items()):
        losses = [rosenbrock(p[0], p[1]) for p in res['path']]
        axes[1].semilogy(losses, color=colors[i], label=name)

    axes[1].set_xlabel('迭代步数')
    axes[1].set_ylabel('损失 (log)')
    axes[1].set_title('损失曲线')
    axes[1].legend()
    axes[1].grid(True, alpha=0.3)

    plt.tight_layout()
    plt.show()

    # 打印最终结果
    print("\n最终损失对比:")
    for name, res in results.items():
        print(f"  {name}: {res['final_loss']:.4f}")

compare_optimizers()

    # ============================================
# 学习率调度器
# ============================================
class LRScheduler:
    """学习率调度器基类"""
    def get_lr(self, epoch):
        raise NotImplementedError

class StepLR(LRScheduler):
    """阶梯衰减：每隔固定 epoch 降低学习率"""
    def __init__(self, initial_lr, step_size, gamma=0.1):
        self.initial_lr = initial_lr
        self.step_size = step_size
        self.gamma = gamma

    def get_lr(self, epoch):
        return self.initial_lr * (self.gamma ** (epoch // self.step_size))

class CosineAnnealing(LRScheduler):
    """余弦退火：按余弦曲线周期性衰减"""
    def __init__(self, initial_lr, T_max, eta_min=0):
        self.initial_lr = initial_lr
        self.T_max = T_max
        self.eta_min = eta_min

    def get_lr(self, epoch):
        return self.eta_min + (self.initial_lr - self.eta_min) * \
               (1 + np.cos(np.pi * epoch / self.T_max)) / 2

class WarmupCosine(LRScheduler):
    """预热 + 余弦退火"""
    def __init__(self, initial_lr, warmup_epochs, total_epochs):
        self.initial_lr = initial_lr
        self.warmup_epochs = warmup_epochs
        self.total_epochs = total_epochs

    def get_lr(self, epoch):
        if epoch < self.warmup_epochs:
            # 线性预热
            return self.initial_lr * epoch / self.warmup_epochs
        else:
            # 余弦退火
            progress = (epoch - self.warmup_epochs) / (self.total_epochs - self.warmup_epochs)
            return self.initial_lr * 0.5 * (1 + np.cos(np.pi * progress))

# ============================================
# 对比不同的学习率调度
# ============================================
epochs = 100

schedulers = {
    '固定 lr=0.01': LRScheduler(),
    '阶梯衰减': StepLR(0.1, 30, 0.1),
    '余弦退火': CosineAnnealing(0.1, epochs, eta_min=1e-5),
    '预热+余弦': WarmupCosine(0.1, 10, epochs),
}

# 画学习率曲线
plt.figure(figsize=(10, 4))
colors = ['blue', 'orange', 'green', 'red']
for (name, sched), color in zip(schedulers.items(), colors):
    lrs = [sched.get_lr(e) for e in range(epochs)]
    plt.plot(lrs, label=name, color=color, linewidth=2)

plt.xlabel('Epoch')
plt.ylabel('学习率')
plt.title('学习率调度对比')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()

print("\n建议：")
print("- 预热(warmup)可以帮助大模型训练初期更稳定")
print("- 余弦退火让学习率缓慢下降，适合精细调优")

    五、关键知识点回顾¶

    优化器 更新规则 优点 适用场景
    SGD θ = θ - α·g 简单高效 大数据集、CNN
    Momentum v = βv + g, θ = θ - αv 加速收敛 峡谷地形
    AdaGrad θ = θ - α·g/√(Σg²) 自适应 稀疏特征
    RMSprop θ = θ - α·g/√(v) 改进AdaGrad 非平稳目标
    Adam 结合Momentum+RMSprop 稳健 默认首选

    六、思考题¶

    1. Adam 有时会泛化不如 SGD+Momentum？ 试着调整 β1, β2 看看效果。

    2. 学习率太大会怎样？ 观察 loss 曲线是否出现 NaN 或发散。

    3. Batch Size 的影响？ 尝试不同的 batch size，观察训练曲线和泛化性能。
