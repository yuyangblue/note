## 一、链式法则的可视化

反向传播的链式法则是深度学习的基础，先来直观理解一下。

```python
import numpy as np
import matplotlib.pyplot as plt

def relu(z):
    return np.maximum(0, z)

def relu_grad(z):
    """ReLU 的导数"""
    return (z > 0).astype(float)

# ============================================
# 链式法则示例：三层网络
# ============================================
"""
假设网络: x -> z1 = W1*x + b1 -> h1 = relu(z1)
       -> z2 = W2*h1 + b2 -> h2 = relu(z2)
       -> z3 = W3*h2 + b3 -> y = z3

损失: L = (y - y_true)² / 2

反向传播：
∂L/∂W3 = ∂L/∂y * ∂y/∂z3 * ∂z3/∂W3 = (y-y_true) * 1 * h2 = (y-y_true) * h2

∂L/∂W2 = ∂L/∂y * ∂y/∂z3 * ∂z3/∂h2 * ∂h2/∂z2 * ∂z2/∂W2
        = (y-y_true) * W3.T * relu_grad(z2) * h1
"""

# 手动计算一个简单例子
np.random.seed(42)

# 假设
x = np.array([1.0])
W1 = np.array([[2.0]])
b1 = np.array([0.5])
W2 = np.array([[3.0]])
b2 = np.array([0.1])
W3 = np.array([[4.0]])
b3 = np.array([0.2])
y_true = 10.0

# 前向传播
z1 = W1 @ x + b1
h1 = relu(z1)
z2 = W2 @ h1 + b2
h2 = relu(z2)
z3 = W3 @ h2 + b3
y = z3

print("前向传播:")
print(f"  x = {x}")
print(f"  z1 = W1*x + b1 = {z1}")
print(f"  h1 = relu(z1) = {h1}")
print(f"  z2 = W2*h1 + b2 = {z2}")
print(f"  h2 = relu(z2) = {h2}")
print(f"  z3 = W3*h2 + b3 = {z3}")
print(f"  y = {y}")
print(f"  损失 L = (y - y_true)²/2 = {(y - y_true)**2/2}")

# 反向传播
dL_dy = y - y_true
dL_dz3 = dL_dy * 1
dL_dW3 = dL_dz3 * h2
dL_db3 = dL_dz3

dL_dh2 = W3.T @ dL_dz3
dL_dz2 = dL_dh2 * relu_grad(z2)
dL_dW2 = dL_dz2 @ h1.T
dL_db2 = dL_dz2

dL_dh1 = W2.T @ dL_dz2
dL_dz1 = dL_dh1 * relu_grad(z1)
dL_dW1 = dL_dz1 @ x.T
dL_db1 = dL_dz1

print("\n反向传播:")
print(f"  ∂L/∂y = {dL_dy}")
print(f"  ∂L/∂W3 = ∂L/∂y * h2 = {dL_dW3}")
print(f"  ∂L/∂W2 = W3.T * ∂L/∂z3 * relu'(z2) * h1 = {dL_dW2}")
print(f"  ∂L/∂W1 = W2.T * ... * relu'(z1) * x = {dL_dW1}")
```

## 二、梯度消失与爆炸实验

这是深度学习最头疼的问题之一！让我们用实验重现它。

```python
# ============================================
# 实验：训练深层网络，观察梯度变化
# ============================================
np.random.seed(42)

def relu(z):
    return np.maximum(0, z)

def relu_grad(z):
    return (z > 0).astype(float)

class DeepNetForGradientTest:
    """用于测试梯度的深层网络"""
    
    def __init__(self, layer_dims, weight_scale=1.0):
        """
        layer_dims: 每层单元数列表
        weight_scale: 权重初始化的缩放因子
        """
        self.W = []
        self.b = []
        self.acts = []  # 保存激活值
        self.zs = []    # 保存预激活值
        
        for i in range(len(layer_dims) - 1):
            fan_in = layer_dims[i]
            fan_out = layer_dims[i+1]
            # 随机初始化
            W = np.random.randn(fan_out, fan_in) * weight_scale / np.sqrt(fan_in)
            b = np.zeros(fan_out)
            self.W.append(W)
            self.b.append(b)
    
    def forward(self, x):
        """前向传播"""
        self.acts = [x]
        self.zs = []
        
        for i in range(len(self.W)):
            z = self.W[i] @ self.acts[-1] + self.b[i]
            self.zs.append(z)
            a = relu(z)
            self.acts.append(a)
        
        return self.acts[-1]
    
    def backward(self, dL_dy):
        """反向传播，返回每层梯度的均值"""
        dL_da = dL_dy
        grad_means = []
        
        for i in reversed(range(len(self.W))):
            # 经过激活函数
            dL_dz = dL_da * relu_grad(self.zs[i])
            
            # 梯度均值（用于分析）
            grad_means.append(np.mean(np.abs(dL_dz)))
            
            # 传向下一层
            dL_da = self.W[i].T @ dL_dz
        
        return grad_means[::-1]  # 从第一层到最后一层

# ============================================
# 测试：不同层数和初始化方差的影响
# ============================================
def test_gradient_flow():
    """测试深层网络的梯度流动"""
    
    layer_dims = [1] + [100] * 20  # 1 -> 100 -> ... -> 100 -> 1
    num_layers = len(layer_dims) - 2  # 隐藏层数量
    
    weight_scales = [0.1, 0.5, 1.0, 2.0, 5.0]
    
    plt.figure(figsize=(12, 5))
    
    # 左图：不同初始化的梯度流
    plt.subplot(1, 2, 1)
    x = np.array([[1.0]])
    y_true = np.array([[10.0]])
    
    for scale in weight_scales:
        net = DeepNetForGradientTest(layer_dims, weight_scale=scale)
        
        # 前向传播
        y_pred = net.forward(x)
        
        # 反向传播
        dL_dy = y_pred - y_true
        grad_means = net.backward(dL_dy)
        
        plt.plot(range(1, num_layers + 1), grad_means, label=f'scale={scale}', marker='o', markersize=3)
    
    plt.xlabel('Layer')
    plt.ylabel('Mean |Gradient|')
    plt.title('梯度流动（不同初始化方差）')
    plt.yscale('log')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.axhline(1.0, color='gray', linestyle='--', alpha=0.5, label='理想值=1')
    
    # 右图：激活值的变化
    plt.subplot(1, 2, 2)
    
    for scale in [0.1, 1.0, 5.0]:
        net = DeepNetForGradientTest(layer_dims, weight_scale=scale)
        net.forward(x)
        
        act_means = [np.mean(np.abs(a)) for a in net.acts[1:]]  # 去掉输入
        plt.plot(range(1, num_layers + 1), act_means, label=f'scale={scale}', marker='o', markersize=3)
    
    plt.xlabel('Layer')
    plt.ylabel('Mean |Activation|')
    plt.title('激活值流动（不同初始化方差）')
    plt.yscale('log')
    plt.legend()
    plt.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()

test_gradient_flow()

print("\n关键观察：")
print("- scale=0.1: 梯度逐层衰减 → 梯度消失")
print("- scale=1.0: 梯度基本稳定 → 良好")
print("- scale=5.0: 梯度逐层爆炸 → 梯度爆炸")
```

## 三、正确的初始化：Xavier 和 He

笔记里提到的初始化方法，我们来实现并验证效果。

```python
# ============================================
# 初始化方法对比
# ============================================

class InitializedNet:
    """使用不同初始化方法的网络"""
    
    def __init__(self, layer_dims, init_method='xavier'):
        self.W = []
        self.b = []
        self.layer_dims = layer_dims
        self.init_method = init_method
        
        for i in range(len(layer_dims) - 1):
            fan_in = layer_dims[i]
            fan_out = layer_dims[i+1]
            
            if init_method == 'zeros':
                W = np.zeros((fan_out, fan_in))
            elif init_method == 'uniform':
                # 均匀分布 [-1/sqrt(fan_in), 1/sqrt(fan_in)]
                W = np.random.uniform(-1, 1, (fan_out, fan_in)) / np.sqrt(fan_in)
            elif init_method == 'xavier':
                # Xavier: 从 N(0, 2/(fan_in+fan_out)) 采样
                std = np.sqrt(2.0 / (fan_in + fan_out))
                W = np.random.randn(fan_out, fan_in) * std
            elif init_method == 'he':
                # He: 从 N(0, 2/fan_in) 采样（适合ReLU）
                std = np.sqrt(2.0 / fan_in)
                W = np.random.randn(fan_out, fan_in) * std
            
            b = np.zeros(fan_out)
            self.W.append(W)
            self.b.append(b)
    
    def forward(self, x):
        self.acts = [x]
        self.zs = []
        
        for i in range(len(self.W)):
            z = self.W[i] @ self.acts[-1] + self.b[i]
            self.zs.append(z)
            a = relu(z)
            self.acts.append(a)
        
        return self.acts[-1]

# ============================================
# 对比不同初始化方法的训练效果
# ============================================
np.random.seed(42)

# 生成数据
N = 200
x_train = np.random.randn(N, 1)
y_train = np.sin(3*x_train.flatten()) * np.exp(-x_train.flatten()**2/4)

# 不同初始化方法
methods = ['zeros', 'uniform', 'xavier', 'he']
results = {}

for method in methods:
    net = InitializedNet([1, 64, 64, 64, 1], init_method=method)
    
    lr = 0.001
    losses = []
    
    for epoch in range(500):
        # 前向
        y_pred = net.forward(x_train.T).T
        
        # 损失
        loss = np.mean((y_pred - y_train)**2)
        losses.append(loss)
        
        # 反向（简化版）
        dL_dy = 2 * (y_pred - y_train) / N
        dL_da = dL_dy
        
        for i in reversed(range(len(net.W))):
            dL_dz = dL_da * (net.zs[i] > 0).astype(float)
            dL_dW = np.outer(dL_dz, net.acts[i])
            net.W[i] -= lr * dL_dW
            net.b[i] -= lr * np.mean(dL_dz, axis=1)
            dL_da = net.W[i].T @ dL_dz
    
    results[method] = losses

# 画图
plt.figure(figsize=(12, 5))

plt.subplot(1, 2, 1)
for method, losses in results.items():
    plt.plot(losses, label=method)
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('不同初始化的训练曲线')
plt.legend()
plt.yscale('log')
plt.grid(True, alpha=0.3)

# 最终拟合效果
plt.subplot(1, 2, 2)
x_test = np.linspace(-3, 3, 200).reshape(-1, 1)
plt.scatter(x_train, y_train, alpha=0.3, s=10, label='数据')

for method in ['xavier', 'he']:
    net = InitializedNet([1, 64, 64, 64, 1], init_method=method)
    # 训练几步看看
    for _ in range(500):
        y_pred = net.forward(x_train.T).T
        dL_dy = 2 * (y_pred - y_train) / N
        dL_da = dL_dy
        for i in reversed(range(len(net.W))):
            dL_dz = dL_da * (net.zs[i] > 0).astype(float)
            dL_dW = np.outer(dL_dz, net.acts[i])
            net.W[i] -= 0.001 * dL_dW
            net.b[i] -= 0.001 * np.mean(dL_dz, axis=1)
            dL_da = net.W[i].T @ dL_dz
    
    y_pred = net.forward(x_test.T).T.flatten()
    plt.plot(x_test, y_pred, label=method, linewidth=2)

plt.legend()
plt.title('拟合效果（Xavier vs He）')
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

print("\n关键结论：")
print("- zeros: 完全不工作，梯度为0")
print("- uniform: 勉强能训练，但很慢")
print("- Xavier: 适合 sigmoid/tanh")
print("- He: 专门为 ReLU 设计，最适合现代网络")
```

## 四、完整的手写反向传播框架

整合前面的知识，写一个完整的神经网络训练框架。

```python
# ============================================
# 完整的神经网络训练框架（NumPy版）
# ============================================
class NeuralNetwork:
    """
    支持任意层数的神经网络
    使用 Xavier/He 初始化和标准反向传播
    """
    
    def __init__(self, layer_dims, activation='relu'):
        """
        layer_dims: [input_dim, hidden1, hidden2, ..., output_dim]
        """
        self.layer_dims = layer_dims
        self.num_layers = len(layer_dims) - 1
        self.activation = activation
        self.params = {}
        
        # 初始化参数
        for i in range(1, len(layer_dims)):
            fan_in = layer_dims[i-1]
            fan_out = layer_dims[i]
            
            if activation == 'relu':
                std = np.sqrt(2.0 / fan_in)
            else:  # tanh, sigmoid
                std = np.sqrt(2.0 / (fan_in + fan_out))
            
            self.params[f'W{i}'] = np.random.randn(fan_out, fan_in) * std
            self.params[f'b{i}'] = np.zeros(fan_out)
    
    def _activate(self, z):
        if self.activation == 'relu':
            return np.maximum(0, z)
        elif self.activation == 'tanh':
            return np.tanh(z)
        elif self.activation == 'sigmoid':
            return 1 / (1 + np.exp(-np.clip(z, -500, 500)))
    
    def _activate_grad(self, z):
        if self.activation == 'relu':
            return (z > 0).astype(float)
        elif self.activation == 'tanh':
            return 1 - np.tanh(z)**2
        elif self.activation == 'sigmoid':
            a = self._activate(z)
            return a * (1 - a)
    
    def forward(self, X):
        """前向传播"""
        self.cache = {'A0': X}
        A = X
        
        for i in range(1, self.num_layers + 1):
            Z = self.params[f'W{i}'] @ A + self.params[f'b{i}']
            if i < self.num_layers:
                A = self._activate(Z)
            else:
                A = Z  # 输出层不做激活
            self.cache[f'Z{i}'] = Z
            self.cache[f'A{i}'] = A
        
        return A
    
    def backward(self, dL_dA):
        """反向传播"""
        m = dL_dA.shape[1]
        grads = {}
        
        for i in reversed(range(1, self.num_layers + 1)):
            A_prev = self.cache[f'A{i-1}']
            Z = self.cache[f'Z{i}']
            W = self.params[f'W{i}']
            
            if i == self.num_layers:
                dL_dZ = dL_dA
            else:
                dL_dZ = dL_dA * self._activate_grad(Z)
            
            grads[f'dW{i}'] = dL_dZ @ A_prev.T / m
            grads[f'db{i}'] = np.mean(dL_dZ, axis=1)
            dL_dA = W.T @ dL_dZ
        
        return grads
    
    def train(self, X, Y, learning_rate=0.01, epochs=100, verbose=100):
        """训练循环"""
        losses = []
        
        for epoch in range(epochs):
            # 前向
            Y_pred = self.forward(X)
            loss = np.mean((Y_pred - Y)**2)
            losses.append(loss)
            
            # 反向
            dL_dY = 2 * (Y_pred - Y) / Y.shape[0]
            grads = self.backward(dL_dY)
            
            # 更新
            for i in range(1, self.num_layers + 1):
                self.params[f'W{i}'] -= learning_rate * grads[f'dW{i}']
                self.params[f'b{i}'] -= learning_rate * grads[f'db{i}']
            
            if epoch % verbose == 0:
                print(f"Epoch {epoch}: loss = {loss:.6f}")
        
        return losses

# ============================================
# 测试框架
# ============================================
np.random.seed(0)

# 数据
X_train = np.random.randn(200, 1)
Y_train = np.sin(3*X_train.flatten()) * np.exp(-X_train.flatten()**2/4)
Y_train = Y_train.reshape(1, -1)

# 创建网络
nn = NeuralNetwork([1, 32, 32, 32, 1], activation='relu')

# 训练
losses = nn.train(X_train.T, Y_train, learning_rate=0.01, epochs=500, verbose=100)

# 测试
X_test = np.linspace(-3, 3, 100).reshape(-1, 1)
Y_test = nn.forward(X_test.T)

plt.figure(figsize=(10, 4))
plt.subplot(1, 2, 1)
plt.plot(losses)
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('训练损失')
plt.yscale('log')

plt.subplot(1, 2, 2)
plt.scatter(X_train, Y_train.T, alpha=0.3, label='训练数据')
plt.plot(X_test, Y_test.T, 'r-', linewidth=2, label='预测')
plt.legend()
plt.title('拟合效果')
plt.show()
```

## 五、关键知识点回顾

| 概念 | 公式 | 要点 |
|------|------|------|
| 链式法则 | ∂L/∂x = ∂L/∂y · ∂y/∂x | 反向传播的基础 |
| 梯度消失 | ∂L/∂W₁ = ∂L/∂y · Π(小梯度) → 0 | 层太深时发生 |
| 梯度爆炸 | ∂L/∂W₁ = ∂L/∂y · Π(大梯度) → ∞ | 初始化太大时发生 |
| Xavier | σ² = 2/(n_in + n_out) | 适合 tanh/sigmoid |
| He | σ² = 2/n_in | 适合 ReLU |

1. **为什么 ReLU 需要 He 而不是 Xavier？** 提示：ReLU 会丢弃一半的信息。
