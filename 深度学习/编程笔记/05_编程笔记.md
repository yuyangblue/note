## 一、常见损失函数一览

先来认识一下各种损失函数的全家福。

```python
import numpy as np
import matplotlib.pyplot as plt

# ============================================
# 1. 回归损失：均方误差 (MSE)
# ============================================
def mse_loss(y_pred, y_true):
    """
    MSE = (1/N) * Σ(y_pred - y_true)²
    对应正态分布假设下的最大似然估计
    """
    return np.mean((y_pred - y_true) ** 2)

def mse_grad(y_pred, y_true):
    """MSE 对预测值的梯度"""
    return 2 * (y_pred - y_true)

# ============================================
# 2. 回归损失：平均绝对误差 (MAE / L1 Loss)
# ============================================
def mae_loss(y_pred, y_true):
    """
    MAE = (1/N) * Σ|y_pred - y_true|
    对异常值更鲁棒，但梯度不连续
    """
    return np.mean(np.abs(y_pred - y_true))

def mae_grad(y_pred, y_true):
    """MAE 的次梯度"""
    return np.sign(y_pred - y_true)

# ============================================
# 3. 二分类损失：二元交叉熵 (BCE)
# ============================================
def bce_loss(y_pred, y_true):
    """
    BCE = -[y*log(σ(z)) + (1-y)*log(1-σ(z))]
    对应 Bernoulli 分布假设
    
    注意：y_pred 需要是概率值 (0, 1)
    """
    # 数值稳定化
    eps = 1e-7
    y_pred = np.clip(y_pred, eps, 1 - eps)
    return -np.mean(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred))

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# ============================================
# 4. 多分类损失：交叉熵 + Softmax
# ============================================
def softmax(z):
    """
    softmax(z_k) = exp(z_k) / Σexp(z_j)
    把原始分数转为概率分布
    """
    # 减去最大值防溢出
    z_shifted = z - np.max(z, axis=-1, keepdims=True)
    exp_z = np.exp(z_shifted)
    return exp_z / np.sum(exp_z, axis=-1, keepdims=True)

def cross_entropy_loss(z, y_true):
    """
    多分类交叉熵损失
    
    z: 网络原始输出 (batch, num_classes)
    y_true: 真实标签 (batch,)，取值为 0, 1, ..., K-1
    
    返回: 平均交叉熵损失
    """
    # 转概率
    probs = softmax(z)
    
    # 取对应类别的概率
    # Numpy 技巧：用 advanced indexing
    batch_size = z.shape[0]
    correct_probs = probs[np.arange(batch_size), y_true]
    
    # 取负对数
    eps = 1e-7
    correct_probs = np.clip(correct_probs, eps, 1)
    
    return -np.mean(np.log(correct_probs))

# ============================================
# 5. Huber 损失（MSE和MAE的折中）
# ============================================
def huber_loss(y_pred, y_true, delta=1.0):
    """
    Huber 损失：
    - 小误差用 MSE
    - 大误差用 MAE（避免梯度爆炸）
    """
    error = y_pred - y_true
    abs_error = np.abs(error)
    
    # 判断是"小误差"还是"大误差"
    is_small = abs_error <= delta
    
    loss = np.where(is_small,
                    0.5 * error ** 2,
                    delta * abs_error - 0.5 * delta ** 2)
    
    return np.mean(loss)

# ============================================
# 6. 对比可视化
# ============================================
# 生成数据：真实值 y_true = 0
y_true = np.zeros(100)
y_pred = np.linspace(-2, 2, 100)

mse = [mse_loss(yp, 0) for yp in y_pred]
mae = [mae_loss(yp, 0) for yp in y_pred]
huber = [huber_loss(yp, 0, delta=1.0) for yp in y_pred]

plt.figure(figsize=(10, 5))
plt.plot(y_pred, mse, label='MSE', linewidth=2)
plt.plot(y_pred, mae, label='MAE', linewidth=2)
plt.plot(y_pred, huber, label='Huber (δ=1)', linewidth=2)
plt.axhline(0, color='gray', linestyle='--', alpha=0.5)
plt.axvline(0, color='gray', linestyle='--', alpha=0.5)
plt.xlabel('y_pred (真实值 y_true = 0)')
plt.ylabel('损失值')
plt.title('回归损失函数对比')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
```

## 二、交叉熵损失的深入理解

二分类交叉熵背后的概率解释——极大似然估计！

```python
# ============================================
# 理解：负对数似然 = 交叉熵
# ============================================
"""
假设我们有一个二分类问题：
- y = 1 的概率是 p
- y = 0 的概率是 1-p

似然函数: L = p^y * (1-p)^(1-y)
对数似然: log L = y*log(p) + (1-y)*log(1-p)
负对数似然: -log L = -y*log(p) - (1-y)*log(1-p)

这正是交叉熵损失！

当用 sigmoid 预测 p = σ(z) 时：
交叉熵 = -y*log(σ(z)) - (1-y)*log(1-σ(z))
"""

# 可视化：损失随预测值的变化
y_true_0 = 0  # 真实标签为0
y_true_1 = 1  # 真实标签为1

z = np.linspace(-5, 5, 200)  # 网络输出（ logits）
p = sigmoid(z)

# 标签0的损失
loss_0 = -np.log(1 - p + 1e-7)  # -log(1-σ(z))
# 标签1的损失
loss_1 = -np.log(p + 1e-7)      # -log(σ(z))

plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.plot(z, loss_0, label='y_true = 0')
plt.plot(z, loss_1, label='y_true = 1')
plt.xlabel('z (网络输出)')
plt.ylabel('损失值')
plt.title('二分类交叉熵损失')
plt.legend()
plt.grid(True, alpha=0.3)

# 右图：概率视角
plt.subplot(1, 2, 2)
plt.plot(p, loss_0, label='y_true = 0')
plt.plot(p, loss_1, label='y_true = 1')
plt.xlabel('p (预测概率)')
plt.ylabel('损失值')
plt.title('损失 vs 预测概率')
plt.legend()
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

# 关键观察：当真实标签是1时，预测0的代价是无穷大！
print("关键观察：")
print(f"当 y_true=1, 预测 p→0 时: loss = {-np.log(1e-7):.2f}")
print("这就是为什么交叉熵对错误预测有强烈的惩罚！")
```

## 三、Softmax 的数值稳定性

softmax 很容易溢出，必须做数值稳定化。

```python
# ============================================
# Softmax 的数值稳定性问题
# ============================================
def softmax_naive(z):
    """不稳定的版本"""
    exp_z = np.exp(z)
    return exp_z / np.sum(exp_z)

def softmax_stable(z):
    """数值稳定的版本"""
    # 减去最大值
    z_shifted = z - np.max(z)
    exp_z = np.exp(z_shifted)
    return exp_z / np.sum(exp_z)

# 测试
z = np.array([1000, 1001, 1002])  # 很大的输入
print("输入:", z)
print("朴素实现:", softmax_naive(z))  # 会溢出：exp(1000) = inf
print("稳定实现:", softmax_stable(z))  # 正常工作

# 再试一个极端例子
z2 = np.array([-1000, -999, -998])  # 很小的输入
print("\n输入:", z2)
try:
    print("朴素实现:", softmax_naive(z2))  # 全部变0
except:
    print("朴素实现: 溢出错误")
print("稳定实现:", softmax_stable(z2))

# ============================================
# 多分类交叉熵的完整实现
# ============================================
def categorical_cross_entropy(logits, labels):
    """
    多分类交叉熵
    
    logits: (batch, num_classes) 原始分数
    labels: (batch,) 整数标签
    
    等价于: -log(softmax(logits)[range(batch), labels])
    """
    probs = softmax(logits)
    
    # 收集正确类别的概率
    batch_idx = np.arange(len(labels))
    correct_probs = probs[batch_idx, labels]
    
    # 负对数
    return -np.mean(np.log(correct_probs + 1e-7))

# 测试
logits = np.array([[2.0, 1.0, 0.1],   # 样本0：预测类别0
                   [0.5, 3.0, 0.2]])  # 样本1：预测类别1
labels = np.array([0, 1])  # 真实标签

print("\n多分类交叉熵测试:")
print(f"logits:\n{logits}")
print(f"labels: {labels}")
print(f"probs:\n{softmax(logits)}")
print(f"交叉熵损失: {categorical_cross_entropy(logits, labels):.4f}")
```

## 四、用损失函数指导训练

把损失函数和梯度下降结合起来。

```python
# ============================================
# 完整示例：二分类任务
# ============================================
np.random.seed(42)

# 生成二分类数据
n_samples = 200
# 两类数据：高斯分布，均值不同
X_class0 = np.random.randn(n_samples, 2) + np.array([-2, -2])
X_class1 = np.random.randn(n_samples, 2) + np.array([2, 2])
X = np.vstack([X_class0, X_class1])
y = np.array([0] * n_samples + [1] * n_samples)

# 画数据
plt.figure(figsize=(8, 6))
plt.scatter(X_class0[:, 0], X_class0[:, 1], c='blue', label='类别0', alpha=0.6)
plt.scatter(X_class1[:, 0], X_class1[:, 1], c='red', label='类别1', alpha=0.6)
plt.xlabel('x1')
plt.ylabel('x2')
plt.title('二分类数据集')
plt.legend()
plt.show()

# ============================================
# 定义网络
# ============================================
class BinaryClassifier:
    def __init__(self, input_dim):
        self.W = np.random.randn(input_dim, 1) * 0.1
        self.b = np.zeros(1)
    
    def forward(self, X):
        z = X @ self.W + self.b
        return sigmoid(z)  # 输出概率
    
    def compute_loss(self, X, y):
        p = self.forward(X)
        return bce_loss(p, y)
    
    def backward(self, X, y):
        """手动反向传播"""
        p = self.forward(X)
        # BCE 对 z 的梯度 = p - y
        dL_dz = (p.flatten() - y).reshape(-1, 1)
        
        dL_dW = X.T @ dL_dz / len(y)
        dL_db = np.mean(dL_dz)
        
        return dL_dW, dL_db
    
    def train(self, X, y, lr=0.1, epochs=100):
        losses = []
        for _ in range(epochs):
            loss = self.compute_loss(X, y)
            dW, db = self.backward(X, y)
            
            self.W -= lr * dW
            self.b -= lr * db
            
            losses.append(loss)
        return losses

# 训练
clf = BinaryClassifier(2)
losses = clf.train(X, y, lr=0.1, epochs=200)

# 画损失曲线
plt.figure(figsize=(10, 4))
plt.subplot(1, 2, 1)
plt.plot(losses)
plt.xlabel('Epoch')
plt.ylabel('BCE Loss')
plt.title('训练损失')
plt.grid(True, alpha=0.3)

# 画决策边界
plt.subplot(1, 2, 2)
h = 0.02
x_min, x_max = X[:, 0].min() - 1, X[:, 0].max() + 1
y_min, y_max = X[:, 1].min() - 1, X[:, 1].max() + 1
xx, yy = np.meshgrid(np.arange(x_min, x_max, h),
                     np.arange(y_min, y_max, h))

Z = clf.forward(np.c_[xx.ravel(), yy.ravel()])
Z = Z.reshape(xx.shape)

plt.contourf(xx, yy, Z, levels=[0, 0.5, 1], colors=['lightblue', 'lightsalmon'], alpha=0.5)
plt.contour(xx, yy, Z, levels=[0.5], colors='black', linewidths=2)
plt.scatter(X_class0[:, 0], X_class0[:, 1], c='blue', label='类别0')
plt.scatter(X_class1[:, 0], X_class1[:, 1], c='red', label='类别1')
plt.xlabel('x1')
plt.ylabel('x2')
plt.title('决策边界')
plt.legend()

plt.tight_layout()
plt.show()

# 测试准确率
predictions = (clf.forward(X) > 0.5).astype(int).flatten()
accuracy = np.mean(predictions == y)
print(f"\n训练准确率: {accuracy:.2%}")
```

## 五、关键知识点回顾

| 损失函数 | 公式 | 适用场景 | 特点 |
|---------|------|---------|------|
| MSE | `(y-ŷ)²` | 回归 | 梯度平滑，对大误差惩罚重 |
| MAE | `|y-ŷ|` | 回归 | 对异常值鲁棒，梯度不连续 |
| BCE | `-ylog(p)-(1-y)log(1-p)` | 二分类 | 对错误惩罚强 |
| CE+Softmax | `-log(σ_k)` | 多分类 | 概率解释，稳定 |
| Huber | MSE+MAE折中 | 回归 | 抗异常值，梯度连续 |
