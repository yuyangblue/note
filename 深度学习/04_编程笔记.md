## 一、从矩阵角度理解层间计算

笔记里用矩阵把"一层"的所有神经元计算打包成一个公式，我们来验证一下。

```python
import numpy as np
import matplotlib.pyplot as plt

def relu(z):
    return np.maximum(0, z)

# 两种写法的对比：标量展开 vs 矩阵形式
print("=" * 50)
print("验证：标量形式 vs 矩阵形式")
print("=" * 50)

# 假设输入 x = [1.0], 有3个隐藏单元
x_scalar = 1.0

# 标量形式（分开写）
theta_0 = np.array([0.5, -0.3, 0.8])  # 偏置 b
theta_1 = np.array([1.0, 2.0, -1.5])   # 权重 w

h1 = relu(theta_0[0] + theta_1[0] * x_scalar)  # 第1个神经元
h2 = relu(theta_0[1] + theta_1[1] * x_scalar)  # 第2个神经元
h3 = relu(theta_0[2] + theta_1[2] * x_scalar)  # 第3个神经元
h_scalar = np.array([h1, h2, h3])

# 矩阵形式（打包计算）
W1 = theta_1.reshape(-1, 1)  # (3, 1)
b1 = theta_0.reshape(-1, 1)  # (3, 1)
h_matrix = relu(W1 @ np.array([[x_scalar]]) + b1).flatten()

print(f"标量形式: h = {h_scalar}")
print(f"矩阵形式: h = {h_matrix}")
print(f"结果一致: {np.allclose(h_scalar, h_matrix)}")
```

## 二、深度网络前向传播实现

现在来搭一个完整的多层网络！

```python
class DeepNeuralNetwork:
    """
    深度神经网络
    
    结构: input -> [Linear -> ReLU] x K -> output
    """
    
    def __init__(self, layer_dims):
        """
        参数:
            layer_dims: 每层的单元数列表
            例如 [input_dim, hidden1, hidden2, ..., output_dim]
        """
        self.layer_dims = layer_dims
        self.num_layers = len(layer_dims) - 1  # 有多少层变换
        
        # 存储所有层的参数
        self.params = {}
        
        # 初始化参数
        for l in range(1, len(layer_dims)):
            # Xavier/He 初始化
            fan_in = layer_dims[l-1]
            fan_out = layer_dims[l]
            # He初始化: std = sqrt(2 / fan_in)
            std = np.sqrt(2 / fan_in)
            
            self.params[f'W{l}'] = np.random.randn(fan_out, fan_in) * std
            self.params[f'b{l}'] = np.zeros(fan_out)
    
    def relu(self, z):
        return np.maximum(0, z)
    
    def relu_grad(self, z):
        return (z > 0).astype(float)
    
    def forward(self, X):
        """
        前向传播
        
        X: (batch_size, input_dim)
        返回: (batch_size, output_dim)
        """
        self.cache = {'A0': X}  # 保存中间结果用于反向传播
        A = X
        
        # 逐层传播
        for l in range(1, self.num_layers + 1):
            W = self.params[f'W{l}']
            b = self.params[f'b{l}']
            
            Z = A @ W.T + b  # (batch, out_dim)
            
            if l == self.num_layers:
                # 最后一层不做激活（回归任务）
                A_next = Z
            else:
                A_next = self.relu(Z)
            
            self.cache[f'Z{l}'] = Z
            self.cache[f'A{l}'] = A_next
            A = A_next
        
        return A
    
    def backward(self, dL_dA):
        """
        反向传播
        
        dL_dA: (batch_size, output_dim)，损失对输出的梯度
        """
        m = dL_dA.shape[0]
        grads = {}
        
        # 从后往前传播
        for l in reversed(range(1, self.num_layers + 1)):
            A = self.cache[f'A{l-1}']
            Z = self.cache[f'Z{l}']
            W = self.params[f'W{l}']
            
            if l == self.num_layers:
                # 输出层：dL/dZ = dL/dA（无激活函数）
                dL_dZ = dL_dA
            else:
                # 隐藏层：乘以ReLU的梯度
                dL_dZ = dL_dA * self.relu_grad(Z)
            
            # 参数梯度
            grads[f'dW{l}'] = (dL_dZ.T @ A) / m  # (out, in)
            grads[f'db{l}'] = np.mean(dL_dZ, axis=0)  # (out,)
            
            # 传给前一层的梯度
            dL_dA = dL_dZ @ W  # (batch, in)
        
        return grads
    
    def update(self, grads, learning_rate):
        """梯度下降更新"""
        for l in range(1, self.num_layers + 1):
            self.params[f'W{l}'] -= learning_rate * grads[f'dW{l}']
            self.params[f'b{l}'] -= learning_rate * grads[f'db{l}']
    
    def predict(self, X):
        return self.forward(X)

# 测试：创建网络
net = DeepNeuralNetwork([1, 16, 16, 16, 1])  # 1-16-16-16-1
print(f"网络结构: {net.layer_dims}")
print(f"层数: {net.num_layers}")

# 数一数参数总量
total_params = 0
for l in range(1, net.num_layers + 1):
    W_shape = net.params[f'W{l}'].shape
    b_shape = net.params[f'b{l}'].shape
    params = W_shape[0] * W_shape[1] + b_shape[0]
    total_params += params
    print(f"Layer {l}: W{W_shape}, b{b_shape}, 参数={params}")
print(f"总参数: {total_params}")
```

## 三、训练一个深度网络拟合复杂函数

用真实数据训练，看看深度网络的能力。

```python
# 生成复杂数据：两个峰的叠加
np.random.seed(42)
N = 200
x_train = np.random.uniform(-2, 2, N)
y_train = np.sin(3*x_train) * np.exp(-x_train**2/4) + np.random.normal(0, 0.1, N)

# 测试数据
x_test = np.linspace(-2, 2, 200)
y_true = np.sin(3*x_test) * np.exp(-x_test**2/4)

# 创建和训练网络
net = DeepNeuralNetwork([1, 32, 32, 32, 1])
loss_fn = lambda y_pred, y_true: np.mean((y_pred - y_true) ** 2)

learning_rate = 0.01
losses = []

for epoch in range(2000):
    # 前向传播
    y_pred = net.forward(x_train.reshape(-1, 1)).flatten()
    
    # 计算损失
    loss = loss_fn(y_pred, y_train)
    losses.append(loss)
    
    # 反向传播
    dL_dA = 2 * (y_pred - y_train) / len(y_train)  # MSE梯度
    grads = net.backward(dL_dA.reshape(-1, 1))
    
    # 更新
    net.update(grads, learning_rate)
    
    if epoch % 500 == 0:
        print(f"Epoch {epoch}: loss = {loss:.6f}")

# 可视化
fig, axes = plt.subplots(1, 2, figsize=(12, 4))

# 损失曲线
axes[0].plot(losses)
axes[0].set_xlabel('Epoch')
axes[0].set_ylabel('Loss')
axes[0].set_title('训练损失')
axes[0].set_yscale('log')

# 拟合效果
axes[1].scatter(x_train, y_train, alpha=0.3, s=20, label='训练数据')
axes[1].plot(x_test, y_true, 'g--', label='真实函数')
y_pred_test = net.predict(x_test.reshape(-1, 1)).flatten()
axes[1].plot(x_test, y_pred_test, 'r-', linewidth=2, label='网络拟合')
axes[1].legend()
axes[1].set_title('拟合效果')
axes[1].set_xlabel('x')
axes[1].set_ylabel('y')

plt.tight_layout()
plt.show()
```

## 四、对比：浅层 vs 深度

一个很宽的浅层网络 vs 一个多层网络，效果一样吗？

```python
# 同样的参数量，不同结构
total_params_target = 500

# 结构1：浅而宽
# 假设: input=1, hidden=H, output=1
# 参数 = H + 1 (W+b), 设为500
H_shallow = total_params_target - 1  # ≈499
net_shallow = DeepNeuralNetwork([1, H_shallow, 1])

# 结构2：窄而深
net_deep = DeepNeuralNetwork([1, 16, 16, 16, 16, 16, 1])

# 计算实际参数
def count_params(net):
    total = 0
    for key in net.params:
        total += net.params[key].size
    return total

print(f"浅层网络: {net_shallow.layer_dims}, 参数量={count_params(net_shallow)}")
print(f"深层网络: {net_deep.layer_dims}, 参数量={count_params(net_deep)}")

# 训练函数
def train_net(net, x, y, epochs=1000, lr=0.01):
    losses = []
    for _ in range(epochs):
        y_pred = net.forward(x.reshape(-1, 1)).flatten()
        loss = np.mean((y_pred - y) ** 2)
        losses.append(loss)
        
        dL = 2 * (y_pred - y) / len(y)
        grads = net.backward(dL.reshape(-1, 1))
        net.update(grads, lr)
    return losses

# 训练
np.random.seed(0)
x_train = np.random.uniform(-2, 2, 100)
y_train = np.sin(3*x_train) * np.exp(-x_train**2/4) + np.random.normal(0, 0.1, 100)

loss_shallow = train_net(net_shallow, x_train, y_train)
loss_deep = train_net(net_deep, x_train, y_train)

# 画对比图
plt.figure(figsize=(12, 4))

plt.subplot(1, 2, 1)
plt.plot(loss_shallow, label='浅而宽')
plt.plot(loss_deep, label='窄而深')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('损失曲线对比')
plt.legend()
plt.yscale('log')

plt.subplot(1, 2, 2)
x_test = np.linspace(-2, 2, 200)
y_true = np.sin(3*x_test) * np.exp(-x_test**2/4)

y_shallow = net_shallow.predict(x_test.reshape(-1, 1)).flatten()
y_deep = net_deep.predict(x_test.reshape(-1, 1)).flatten()

plt.scatter(x_train, y_train, alpha=0.3, s=20, label='数据')
plt.plot(x_test, y_true, 'g--', alpha=0.5, label='真实')
plt.plot(x_test, y_shallow, label=f'浅层 ({count_params(net_shallow)}参数)')
plt.plot(x_test, y_deep, label=f'深层 ({count_params(net_deep)}参数)')
plt.legend()
plt.title('拟合效果对比')

plt.tight_layout()
plt.show()
```

## 六、关键知识点回顾

| 概念 | 代码实现 | 要点 |
|------|---------|------|
| 矩阵表示 | `Z = A @ W.T + b` | 一行代码计算整层 |
| 前向传播 | 逐层缓存 Z, A | 反向传播需要这些中间结果 |
| 反向传播 | 链式法则反向传递 | `dL/dW = dL/dZ @ A.T` |
| 参数更新 | `W = W - lr * dW` | 梯度下降 |
| 深度 vs 宽度 | 不同结构，同等参数 | 深度网络通常更高效 |
