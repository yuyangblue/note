    import numpy as np
import matplotlib.pyplot as plt

# 定义各种激活函数
def relu(z):
    """ReLU: max(0, z)"""
    return np.maximum(0, z)

def leaky_relu(z, alpha=0.1):
    """Leaky ReLU: max(alpha*z, z)"""
    return np.where(z > 0, z, alpha * z)

def sigmoid(z):
    """Sigmoid: 1/(1+exp(-z))"""
    return 1 / (1 + np.exp(-z))

def tanh(z):
    """Tanh: (exp(z)-exp(-z))/(exp(z)+exp(-z))"""
    return np.tanh(z)

def softmax(z):
    """Softmax: exp(z)/sum(exp(z))"""
    # 减去最大值提高数值稳定性
    z_shifted = z - np.max(z)
    return np.exp(z_shifted) / np.sum(np.exp(z_shifted))

def gelu(z):
    """GELU: z * Phi(z)，近似形式"""
    # 精确版本需要计算标准正态分布的CDF
    # 这里用近似: 0.5*z*(1+tanh(sqrt(2/pi)*(z+0.044715*z^3)))
    return 0.5 * z * (1 + np.tanh(np.sqrt(2/np.pi) * (z + 0.044715*z**3)))

def silu_swish(z):
    """SiLU (Swish): z * sigmoid(z)"""
    return z * sigmoid(z)

# 创建输入范围
z = np.linspace(-5, 5, 500)

# 激活函数字典
activations = {
    'ReLU': relu(z),
    'Leaky ReLU (α=0.1)': leaky_relu(z),
    'Sigmoid': sigmoid(z),
    'Tanh': tanh(z),
    'GELU': gelu(z),
    'Swish': silu_swish(z)
}

# 画图
fig, axes = plt.subplots(2, 3, figsize=(12, 7))
axes = axes.flatten()

for ax, (name, a) in zip(axes, activations.items()):
    ax.plot(z, a, 'b-', linewidth=2)
    ax.axhline(0, color='gray', linestyle='--', alpha=0.5)
    ax.axvline(0, color='gray', linestyle='--', alpha=0.5)
    ax.set_title(name, fontsize=12)
    ax.set_xlabel('z')
    ax.set_ylabel('a(z)')
    ax.set_ylim(-1.5, 2)
    ax.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()

    为什么这样写？ - 用 np.where 实现 Leaky ReLU，避免索引问题 - Softmax 要做数值稳定化（减去最大值），防止 exp 溢出 - GELU 的近似公式避免了需要计算误差函数的麻烦

    class ShallowNeuralNet:
    """
    单隐藏层神经网络

    结构: 输入 -> 隐藏层(ReLU) -> 输出层(线性)
    """

    def __init__(self, input_dim, hidden_units):
        """
        参数:
            input_dim: 输入维度
            hidden_units: 隐藏层神经元数量
        """
        self.input_dim = input_dim
        self.hidden_units = hidden_units

        # 初始化参数（随机小值）
        # 注意：实际应该用 Xavier/He 初始化，这里简化处理
        scale_w = 0.1
        self.W1 = np.random.randn(hidden_units, input_dim) * scale_w  # (hidden, input)
        self.b1 = np.zeros(hidden_units)  # (hidden,)

        self.W2 = np.random.randn(1, hidden_units) * scale_w  # (1, hidden)
        self.b2 = np.zeros(1)  # (1,)

    def forward(self, x):
        """
        前向传播

        x: (batch_size, input_dim) 或 (input_dim,)
        返回: (batch_size, 1) 或 (1,)
        """
        # 确保是二维数组
        x = np.atleast_2d(x)

        # 隐藏层: h = ReLU(W1 @ x + b1)
        self.z1 = self.W1 @ x.T + self.b1.reshape(-1, 1)  # (hidden, batch)
        self.h = relu(self.z1)  # (hidden, batch)

        # 输出层: y = W2 @ h + b2
        self.z2 = self.W2 @ self.h + self.b2.reshape(-1, 1)  # (1, batch)

        # 如果输入是1D，返回1D输出
        if x.shape[0] == 1:
            return self.z2.flatten()[0]
        return self.z2.T  # (batch, 1)

    def parameters(self):
        """返回所有参数"""
        return {
            'W1': self.W1, 'b1': self.b1,
            'W2': self.W2, 'b2': self.b2
        }

# 测试
net = ShallowNeuralNet(input_dim=1, hidden_units=3)

# 画网络结构
x = np.linspace(-1, 1, 100)
y = net.forward(x)

plt.figure(figsize=(10, 4))
plt.subplot(1, 2, 1)
plt.scatter(x, y, s=10)
plt.title('随机初始化后的网络输出')
plt.xlabel('x')
plt.ylabel('y')

# 打印参数

print("网络结构: 1 -> 3 -> 1")

    def mse_loss(y_pred, y_true):
    """均方误差损失"""
    return np.mean((y_pred - y_true) ** 2)

def mse_grad(y_pred, y_true):
    """MSE 损失对输出的梯度"""
    return 2 * (y_pred - y_true) / len(y_pred)

class ShallowNetTrainer:
    """带训练的浅层网络"""

    def __init__(self, input_dim, hidden_units, lr=0.01):
        self.net = ShallowNeuralNet(input_dim, hidden_units)
        self.lr = lr

    def train_step(self, x, y_true):
        """一次训练步骤"""
        # 前向传播（保存中间结果）
        y_pred = self.net.forward(x)

        # 反向传播（手动计算梯度）
        batch_size = len(x)

        # 输出层梯度
        dL_dz2 = mse_grad(y_pred.flatten(), y_true)  # (batch,)

        # W2, b2 梯度
        dL_dW2 = np.outer(dL_dz2, self.net.h)  # (1, hidden)
        dL_db2 = dL_dz2  # (batch,) -> mean later

        # 隐藏层梯度（反向传播经过ReLU）
        dL_dh = self.net.W2.T * dL_dz2  # (hidden, batch)
        dL_dz1 = dL_dh * (self.net.z1 > 0).astype(float)  # ReLU梯度

        # W1, b1 梯度
        dL_dW1 = dL_dz1 @ np.atleast_2d(x)  # (hidden, input)
        dL_db1 = np.mean(dL_dz1, axis=1)  # (hidden,)

        # 梯度下降更新
        self.net.W2 -= self.lr * np.mean(dL_dW2, axis=0, keepdims=True)
        self.net.b2 -= self.lr * np.mean(dL_db2)
        self.net.W1 -= self.lr * dL_dW1
        self.net.b1 -= self.lr * dL_db1

        return mse_loss(y_pred.flatten(), y_true)

    def fit(self, x, y, epochs=1000, verbose=100):
        """训练循环"""
        losses = []
        for epoch in range(epochs):
            loss = self.train_step(x, y)
            losses.append(loss)

            if epoch % verbose == 0:
                print(f"Epoch {epoch}: loss = {loss:.6f}")

        return losses

# 训练示例：拟合 sin(2πx)
np.random.seed(0)
x_train = np.random.uniform(0, 1, 50)
y_train = np.sin(2 * np.pi * x_train) + np.random.normal(0, 0.1, 50)

# 创建并训练网络
trainer = ShallowNetTrainer(input_dim=1, hidden_units=20, lr=0.1)
losses = trainer.fit(x_train, y_train, epochs=2000, verbose=500)

# 画损失曲线
plt.figure(figsize=(10, 4))
plt.subplot(1, 2, 1)
plt.plot(losses)
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('训练损失曲线')
plt.yscale('log')

# 画拟合结果
plt.subplot(1, 2, 2)
x_test = np.linspace(0, 1, 200)
y_pred = trainer.net.forward(x_test)
plt.scatter(x_train, y_train, alpha=0.5, label='训练数据')
plt.plot(x_test, y_pred, 'r-', label='网络预测')
plt.plot(x_test, np.sin(2*np.pi*x_test), 'g--', alpha=0.5, label='真实函数')
plt.legend()
plt.title('拟合效果')
plt.tight_layout()
plt.show()

    为什么这样写？ - 反向传播用链式法则，一层层把梯度传回去 - ReLU 的梯度是 z > 0 时为 1，否则为 0 - 学习率 lr=0.1 对于这个小例子足够，太大容易震荡

    六、关键知识点回顾¶

    概念 代码实现 要点
    ReLU np.maximum(0, z) 简单高效，但负值单元会"死"
    前向传播 h = ReLU(W1@x + b1), y = W2@h + b2 逐层计算，保存中间结果
    反向传播 链式法则 + 梯度反向传递 ReLU 梯度 = (z>0)
    参数更新 W = W - lr * dL/dW 梯度下降
    万能逼近 分段线性组合 ReLU 让网络成为分段线性函数

    七、思考题¶

    1. Leaky ReLU 解决了什么问题？ 试着把代码里的 relu 换成 leaky_relu，看看"死亡 ReLU"问题是否改善。

    2. 隐藏单元越多越好吗？ 对比 5、20、100 个隐藏单元的训练效果和泛化能力。

    3. 为什么网络输出是分段线性的？ 数一数你的网络把 x 轴分成了几段。
