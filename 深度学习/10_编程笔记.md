## 一、卷积操作：从手动到理解

卷积就是"滑动窗口"式的加权求和。

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy import signal

# ============================================
# 一维卷积手动实现
# ============================================
def conv1d_manual(x, kernel, stride=1, padding=0):
    """
    手动实现一维卷积
    
    参数:
        x: 输入信号 (length,)
        kernel: 卷积核 (kernel_size,)
        stride: 步长
        padding: 填充数量
    
    返回:
        输出信号
    """
    # 填充
    if padding > 0:
        x = np.pad(x, (padding, padding), mode='constant')
    
    length = len(x)
    k_size = len(kernel)
    
    # 计算输出长度
    out_length = (length - k_size) // stride + 1
    
    # 滑动窗口计算
    result = np.zeros(out_length)
    for i in range(out_length):
        start = i * stride
        result[i] = np.sum(kernel * x[start:start + k_size])
    
    return result

# 测试用例
x = np.array([1, 2, 3, 4, 5, 6, 7, 8])
kernel = np.array([1, 0, -1])  # 边缘检测核

# 手动实现
result_manual = conv1d_manual(x, kernel)

# scipy 验证
result_scipy = signal.convolve(x, kernel, mode='valid')

print("一维卷积验证:")
print(f"  输入:      {x}")
print(f"  卷积核:    {kernel}")
print(f"  手动实现:  {result_manual}")
print(f"  scipy:     {result_scipy}")
print(f"  结果一致:  {np.allclose(result_manual, result_scipy)}")

# ============================================
# 二维卷积手动实现
# ============================================
def conv2d_manual(image, kernel, stride=1, padding=0):
    """
    手动实现二维卷积（单通道）
    
    参数:
        image: 输入图像 (H, W)
        kernel: 卷积核 (K, K)
        stride: 步长
        padding: 填充
    
    返回:
        输出特征图 (H', W')
    """
    # 填充
    if padding > 0:
        image = np.pad(image, padding, mode='constant')
    
    H, W = image.shape
    K = kernel.shape[0]
    
    # 输出尺寸
    out_H = (H - K) // stride + 1
    out_W = (W - K) // stride + 1
    
    # 结果
    result = np.zeros((out_H, out_W))
    
    for i in range(out_H):
        for j in range(out_W):
            h_start = i * stride
            w_start = j * stride
            result[i, j] = np.sum(kernel * image[h_start:h_start+K, w_start:w_start+K])
    
    return result

# 创建测试图像
np.random.seed(42)
image = np.random.rand(8, 8)
kernel = np.array([[1, 0, -1],
                   [2, 0, -2],
                   [1, 0, -1]])  # Sobel 边缘检测

result = conv2d_manual(image, kernel)

# 可视化
plt.figure(figsize=(12, 4))

plt.subplot(1, 3, 1)
plt.imshow(image, cmap='gray')
plt.title('输入图像')
plt.colorbar()

plt.subplot(1, 3, 2)
plt.imshow(kernel, cmap='coolwarm')
plt.title('卷积核 (边缘检测)')
plt.colorbar()

plt.subplot(1, 3, 3)
plt.imshow(result, cmap='gray')
plt.title('输出特征图')
plt.colorbar()

plt.tight_layout()
plt.show()

print(f"\n二维卷积:")
print(f"  输入: {image.shape}, 卷积核: {kernel.shape}")
print(f"  输出: {result.shape}")
```

## 三、用 PyTorch 搭建 CNN

工业级实现！手写数字识别走起。

```python
# ============================================
# PyTorch CNN 完整实现
# ============================================
try:
    import torch
    import torch.nn as nn
    import torch.optim as optim
    from torch.utils.data import DataLoader, TensorDataset
    from torchvision import datasets, transforms
    
    HAS_TORCH = True
except ImportError:
    print("PyTorch 未安装，跳过 CNN 部分")
    HAS_TORCH = False

if HAS_TORCH:
    # ============================================
    # 定义 CNN 模型
    # ============================================
    class SimpleCNN(nn.Module):
        """
        简单卷积神经网络
        
        结构: Conv -> ReLU -> Pool -> Conv -> ReLU -> Pool -> FC
        """
        
        def __init__(self, num_classes=10):
            super(SimpleCNN, self).__init__()
            
            # 第一个卷积块
            # 输入: (batch, 1, 28, 28)
            self.conv1 = nn.Conv2d(1, 16, kernel_size=3, padding=1)  # -> (batch, 16, 28, 28)
            self.relu1 = nn.ReLU()
            self.pool1 = nn.MaxPool2d(kernel_size=2, stride=2)       # -> (batch, 16, 14, 14)
            
            # 第二个卷积块
            self.conv2 = nn.Conv2d(16, 32, kernel_size=3, padding=1)  # -> (batch, 32, 14, 14)
            self.relu2 = nn.ReLU()
            self.pool2 = nn.MaxPool2d(kernel_size=2, stride=2)       # -> (batch, 32, 7, 7)
            
            # 全连接层
            self.fc = nn.Linear(32 * 7 * 7, num_classes)
        
        def forward(self, x):
            # 卷积块1
            x = self.conv1(x)
            x = self.relu1(x)
            x = self.pool1(x)
            
            # 卷积块2
            x = self.conv2(x)
            x = self.relu2(x)
            x = self.pool2(x)
            
            # 展平 + 全连接
            x = x.view(x.size(0), -1)
            x = self.fc(x)
            
            return x
    
    # ============================================
    # 打印网络结构
    # ============================================
    model = SimpleCNN()
    print("CNN 结构:")
    print(model)
    
    # 计算参数量
    total_params = sum(p.numel() for p in model.parameters())
    trainable_params = sum(p.numel() for p in model.parameters() if p.requires_grad)
    print(f"\n参数量:")
    print(f"  总参数: {total_params:,}")
    print(f"  可训练: {trainable_params:,}")
    
    # ============================================
    # 各层的感受野计算
    # ============================================
    print("\n感受野变化:")
    print("  输入: 28x28")
    print("  Conv1 (k=3, p=1): 28x28 -> 28x28")
    print("  Pool1 (2x2): 28x28 -> 14x14 (感受野: 2x2)")
    print("  Conv2 (k=3, p=1): 14x14 -> 14x14")
    print("  Pool2 (2x2): 14x14 -> 7x7 (感受野: 4x4)")
    print("  最终: 每个输出像素对应输入的 4x4 区域")
```

## 五、完整训练流程（MNIST）

从数据加载到训练到评估，一条龙服务。

```python
if HAS_TORCH:
    # ============================================
    # 完整训练 MNIST
    # ============================================
    print("\n" + "="*50)
    print("训练 MNIST 手写数字识别")
    print("="*50)
    
    # 设备
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    print(f"使用设备: {device}")
    
    # 数据预处理
    transform = transforms.Compose([
        transforms.ToTensor(),  # 转为 [0,1] 的 Tensor
        transforms.Normalize((0.1307,), (0.3081,))  # MNIST 的均值和标准差
    ])
    
    # 加载数据
    try:
        train_dataset = datasets.MNIST('./data', train=True, download=True, transform=transform)
        test_dataset = datasets.MNIST('./data', train=False, transform=transform)
        
        train_loader = DataLoader(train_dataset, batch_size=64, shuffle=True)
        test_loader = DataLoader(test_dataset, batch_size=1000, shuffle=False)
        
        print(f"训练集: {len(train_dataset)} 样本")
        print(f"测试集: {len(test_dataset)} 样本")
        
        # 创建模型
        model = SimpleCNN().to(device)
        criterion = nn.CrossEntropyLoss()
        optimizer = optim.Adam(model.parameters(), lr=0.001)
        
        # 训练函数
        def train_epoch(model, loader, criterion, optimizer, device):
            model.train()
            total_loss = 0
            correct = 0
            total = 0
            
            for batch_idx, (data, target) in enumerate(loader):
                data, target = data.to(device), target.to(device)
                
                optimizer.zero_grad()
                output = model(data)
                loss = criterion(output, target)
                loss.backward()
                optimizer.step()
                
                total_loss += loss.item()
                pred = output.argmax(dim=1)
                correct += pred.eq(target).sum().item()
                total += target.size(0)
            
            return total_loss / len(loader), 100. * correct / total
        
        # 测试函数
        def test(model, loader, device):
            model.eval()
            correct = 0
            total = 0
            
            with torch.no_grad():
                for data, target in loader:
                    data, target = data.to(device), target.to(device)
                    output = model(data)
                    pred = output.argmax(dim=1)
                    correct += pred.eq(target).sum().item()
                    total += target.size(0)
            
            return 100. * correct / total
        
        # 训练循环
        epochs = 5
        for epoch in range(1, epochs + 1):
            train_loss, train_acc = train_epoch(model, train_loader, criterion, optimizer, device)
            test_acc = test(model, test_loader, device)
            print(f"Epoch {epoch}: Loss={train_loss:.4f}, Train Acc={train_acc:.2f}%, Test Acc={test_acc:.2f}%")
        
        print(f"\n最终测试准确率: {test_acc:.2f}%")
        
    except Exception as e:
        print(f"数据加载失败: {e}")
        print("这是正常的，首次运行需要下载数据集")
```

## 六、关键 CNN 概念回顾

| 概念 | 公式/代码 | 要点 |
|------|----------|------|
| 卷积 | `y[i,j] = Σk w[k] * x[i+k, j+l]` | 局部连接 |
| 步长 | stride=2 | 输出尺寸减半 |
| 填充 | padding=1 | 保持尺寸 |
| 池化 | MaxPool2d(2,2) | 下采样，保留显著特征 |
| 感受野 | 逐层增大 | 最终特征对应原图多大区域 |
