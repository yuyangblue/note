## 📐 示例2：手动实现最小二乘法

```python
def manual_linear_regression(X, y):
    """
    手动实现最小二乘法
    
    公式: w = (X^T X)^(-1) X^T y
    """
    # 添加偏置项（如果X没有）
    if X.shape[1] == 1 or not np.all(X[:, 0] == 1):
        X_b = np.c_[np.ones((X.shape[0], 1)), X]
    else:
        X_b = X
    
    # 使用正规方程求解
    XtX = X_b.T @ X_b
    XtX_inv = np.linalg.inv(XtX)
    w = XtX_inv @ X_b.T @ y
    
    return w

# 测试手动实现
np.random.seed(42)
X_test_m = 2 * np.random.rand(100, 1)
y_test_m = 4 + 3 * X_test_m + np.random.randn(100, 1)

w_manual = manual_linear_regression(X_test_m, y_test_m)

print("手动实现最小二乘法结果：")
print(f"权重向量 w: {w_manual.flatten()}")
print(f"解释: w[0] = {w_manual[0][0]:.4f} (截距), w[1:] = {w_manual[1:].flatten()} (系数)")

# 对比sklearn结果
sklearn_model = LinearRegression()
sklearn_model.fit(X_test_m, y_test_m)
print(f"\nsklearn结果对比：")
print(f"截距: {sklearn_model.intercept_[0]:.4f}")
print(f"系数: {sklearn_model.coef_[0][0]:.4f}")
```

```python
def fit_polynomial(X_train, y_train, X_test, y_test, degrees):
    """
    拟合不同阶数的多项式
    """
    results = []
    
    plt.figure(figsize=(15, 10))
    
    for i, degree in enumerate(degrees):
        # 创建多项式特征
        poly = PolynomialFeatures(degree=degree)
        X_train_poly = poly.fit_transform(X_train)
        X_test_poly = poly.transform(X_test)
        
        # 训练模型
        model = LinearRegression()
        model.fit(X_train_poly, y_train)
        
        # 预测
        y_train_pred = model.predict(X_train_poly)
        y_test_pred = model.predict(X_test_poly)
        
        # 计算指标
        train_mse = mean_squared_error(y_train, y_train_pred)
        test_mse = mean_squared_error(y_test, y_test_pred)
        train_r2 = r2_score(y_train, y_train_pred)
        test_r2 = r2_score(y_test, y_test_pred)
        
        results.append({
            'degree': degree,
            'train_mse': train_mse,
            'test_mse': test_mse,
            'train_r2': train_r2,
            'test_r2': test_r2,
            'n_params': X_train_poly.shape[1]
        })
        
        # 可视化
        plt.subplot(2, 3, i + 1)
        
        # 生成平滑曲线
        X_plot = np.linspace(0, 4 * np.pi, 200).reshape(-1, 1)
        X_plot_poly = poly.transform(X_plot)
        y_plot = model.predict(X_plot_poly)
        
        plt.scatter(X_train, y_train, c='blue', alpha=0.5, s=20, label='Train')
        plt.scatter(X_test, y_test, c='green', alpha=0.8, s=30, label='Test')
        plt.plot(X_plot, y_plot, 'r-', linewidth=2, label='Fitted')
        plt.plot(X_plot, np.sin(X_plot), 'k--', linewidth=1, alpha=0.5, label='True')
        plt.ylim(-1.5, 1.5)
        plt.title(f'Degree {degree}\nTrain R²={train_r2:.3f}, Test R²={test_r2:.3f}')
        plt.legend(fontsize=8)
        plt.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()
    
    return results

# 划分数据
X_train_nl, X_test_nl, y_train_nl, y_test_nl = train_test_split(
    X_nl, y_nl, test_size=0.3, random_state=42

degrees = [1, 3, 5, 7, 10, 15]

    print(f"{r['degree']:<8} {r['n_params']:<10} {r['train_mse']:<12.4f} {r['test_mse']:<12.4f} {r['train_r2']:<12.4f} {r['test_r2']:<10.4f}")
```

## 📉 示例4：学习曲线可视化

```python
def plot_learning_curve(X, y, model_class, model_params, train_sizes):
    """
    绘制学习曲线
    """
    from sklearn.model_selection import learning_curve
    
    train_sizes_abs, train_scores, test_scores = learning_curve(
        model_class(**model_params), X, y,
        train_sizes=train_sizes,
        cv=5,
        scoring='neg_mean_squared_error',
        n_jobs=-1
    )
    
    train_scores = -train_scores  # 转回正值
    test_scores = -test_scores
    
    train_mean = train_scores.mean(axis=1)
    train_std = train_scores.std(axis=1)
    test_mean = test_scores.mean(axis=1)
    test_std = test_scores.std(axis=1)
    
    plt.figure(figsize=(10, 6))
    
    plt.fill_between(train_sizes_abs, train_mean - train_std, 
                     train_mean + train_std, alpha=0.1, color='blue')
    plt.fill_between(train_sizes_abs, test_mean - test_std, 
                     test_mean + test_std, alpha=0.1, color='orange')
    
    plt.plot(train_sizes_abs, train_mean, 'o-', color='blue', 
             label='Training error')
    plt.plot(train_sizes_abs, test_mean, 'o-', color='orange', 
             label='Cross-validation error')
    
    plt.xlabel('Training examples')
    plt.ylabel('MSE')
    plt.title('Learning Curve')
    plt.legend(loc='best')
    plt.grid(True, alpha=0.3)
    plt.show()
    
    return train_mean, test_mean

# 为不同模型绘制学习曲线
plt.figure(figsize=(15, 5))
```

## 🔬 示例5：偏差-方差分解可视化

```python
def bias_variance_demo():
    """
    偏差方差分解演示
    """
    np.random.seed(42)
    
    # 真实函数
    def true_function(x):
        return np.sin(x)
    
    # 生成多个训练集
    n_datasets = 100
    n_samples = 25
    degrees = [1, 3, 5, 15]
    
    fig, axes = plt.subplots(2, 2, figsize=(14, 10))
    
    for ax, degree in zip(axes.flat, degrees):
        # 收集预测结果
        all_predictions = []
        
        for _ in range(n_datasets):
            # 随机采样
            X_sample = np.sort(2 * np.pi * np.random.rand(n_samples, 1))
            y_sample = true_function(X_sample) + 0.3 * np.random.randn(n_samples, 1)
            
            # 拟合
            poly = PolynomialFeatures(degree=degree)
            X_poly = poly.fit_transform(X_sample)
            model = LinearRegression()
            model.fit(X_poly, y_sample)
            
            # 预测
            X_plot = np.linspace(0, 2 * np.pi, 100).reshape(-1, 1)
            X_plot_poly = poly.transform(X_plot)
            y_pred = model.predict(X_plot_poly)
            all_predictions.append(y_pred.flatten())
        
        all_predictions = np.array(all_predictions)
        mean_prediction = all_predictions.mean(axis=0)
        
        # 计算偏差和方差
        true_values = true_function(X_plot.flatten())
        bias_squared = (mean_prediction - true_values) ** 2
        variance = all_predictions.var(axis=0)
        
        # 可视化
        ax.fill_between(X_plot.flatten(), 
                        mean_prediction - np.sqrt(variance),
                        mean_prediction + np.sqrt(variance),
                        alpha=0.3, color='blue', label='±1 Std (Variance)')
        ax.plot(X_plot.flatten(), mean_prediction, 'b-', linewidth=2, 
                label='Mean Prediction')
        ax.plot(X_plot.flatten(), true_values, 'r--', linewidth=2, 
                label='True Function')
        
        # 绘制一些样本预测
        for i in range(5):
            ax.plot(X_plot.flatten(), all_predictions[i], 'g-', 
                    alpha=0.1, linewidth=1)
        
        ax.set_title(f'Degree {degree}\n'
                    f'Bias² = {bias_squared.mean():.4f}, '
                    f'Var = {variance.mean():.4f}')
        ax.legend()
        ax.set_ylim(-2, 2)
        ax.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()
    
    print("\n偏差-方差权衡总结：")
    print("-" * 50)
    print("Degree 1: 高偏差 + 低方差 → 欠拟合")
    print("Degree 3: 较低偏差 + 较低方差 → 较好")
    print("Degree 5: 低偏差 + 低方差 → 最佳")
    print("Degree 15: 极低偏差 + 高方差 → 过拟合")

bias_variance_demo()
```

**关键洞察**：
- **低阶多项式（欠拟合）**：偏差大，方差小
- **高阶多项式（过拟合）**：偏差小，方差大
- **最佳模型**：偏差和方差达到平衡

## 🎓 总结

```python
"""
回归分析基本概念 编程要点总结：
================================

1. 线性回归模型:
   - 模型: y = Xw + b
   - 求解: w = (X^T X)^(-1) X^T y (正规方程)
   - sklearn: LinearRegression()

2. 多项式回归:
   - PolynomialFeatures(degree=n) 生成多项式特征
   - 本质上仍是线性回归（对参数而言）

3. 过拟合 vs 欠拟合:
   - 欠拟合: 模型太简单，增加复杂度
   - 过拟合: 模型太复杂，使用正则化或更多数据

4. 评估指标:
   - MSE: 均方误差，越小越好
   - R²: 决定系数，0-1之间，越接近1越好

5. 学习曲线:
   - 训练样本增加时的训练/验证误差变化
   - 用于诊断模型问题

6. 偏差-方差分解:
   - 总误差 = 偏差² + 方差 + 不可约误差
   - 目标: 找到最佳平衡点
"""
```
